\chapter*{Devoir surveille -- Taux d'evolution, pourcentages et informations chiffrees}

\begin{enumerate}
\item Une association compte $420$ adherents. Parmi eux, $147$ ont moins de $25$ ans.
\begin{enumerate}
\item Calculer la proportion d'adherents de moins de $25$ ans.
\item Exprimer cette proportion en pourcentage.
\end{enumerate}

\item Un ordinateur coute $850$ \euro{}. Son prix baisse de $12\pourcent$ pendant les soldes.
\begin{enumerate}
\item Donner le coefficient multiplicateur associe a cette baisse.
\item Calculer le nouveau prix de l'ordinateur.
\end{enumerate}

\item Le nombre de visiteurs d'un site internet passe de $18\,000$ a $22\,500$ en un mois.
\begin{enumerate}
\item Calculer la variation absolue.
\item Calculer le taux d'evolution en pourcentage.
\end{enumerate}

\item Un produit subit deux evolutions successives : une hausse de $30\pourcent$, puis une baisse de $20\pourcent$.
\begin{enumerate}
\item Calculer le coefficient multiplicateur global.
\item Donner le taux d'evolution global.
\item Un prix initial de $50$ \euro{} devient-il superieur ou inferieur au prix initial apres ces deux evolutions ?
\end{enumerate}

\item Apres une hausse de $25\pourcent$, un prix vaut $150$ \euro{}. Calculer le prix avant la hausse.
\end{enumerate}
