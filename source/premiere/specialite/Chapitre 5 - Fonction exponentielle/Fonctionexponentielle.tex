\chapter{La fonction exponentielle}

On cherche les fonctions $f$ dérivables sur $\R$ telles que 
$\displaystyle \left\{
\begin{array}{rcl} f(0)&=&1\\
f'&=&f
\end{array}
\right.$

\section{Définition de la fonction exponentielle}
\subsection{Théorčme et définition}

\begin{thm}\label{cauchy-lip}
Il {\em existe} (admis) une {\em unique} fonction $f$ définie et dérivable sur $\R$ telle que
$f(0)=1$ et $f'=f$. 

Cette fonction est appelée exponentielle et notée $\exp$.
\end{thm}

\subsection{Unicité}

Pour démontrer l'unicité d'une telle fonction nous allons avoir besoin d'un
résultat intermédiaire :

\begin{lemme}
Si $f$ est une fonction dérivable sur $\R$ telle que $f(0)=1$ et $f'=f$, alors
\begin{enumerate}
\item Pour tout $x \in \R$, $f(x) \times f(-x) = 1$.
\item La fonction $f$ ne s'annule pas sur $\R$, elle est męme strictement positive.
\end{enumerate}
\end{lemme}

\begin{demo}[ Lemme 1 ]
\begin{enumerate}
\item Pour tout $x\in \R$, on pose $\phi(x) = f(x) f(-x)$. La fonction $\phi$
  est définie et dérivable sur $\R$. De plus $\phi$ est un produit de la forme
  $u \times v$, oů $v$ est une fonction composée:
\[ \phi'(x) = f'(x) f(-x) + f(x) f'(-x) \times (-1) = f(x) f(-x) - f(x) f(-x)
= 0\]
 Ainsi $\phi$ est constante. Or $\phi(0) = f(0) f(-0) = 1 \times 1 = 1$. Donc,
 pour tout $x \in \R$, 
\[ f(x) f(-x) = \phi(x) = \phi(0) = 1\]
\item \begin{itemize}
\item[$\bullet$] Raisonnons par l'absurde : soit $a\in \R$ tel que $f(a) = 0$.
Alors $\phi(a) = f(a) f(-a) = 0 \times f(-a) = 0$. Mais, d'aprčs ci-dessus,
$\phi(a) = 1$. Ce qui est absurde.

\item[$\bullet$] Pour montrer que $f>0$, raisonnons aussi par l'absurde : soit $a\in \R$ tel
que $f(a) <0$. 

La fonction $f$ est dérivable donc continue sur $\R$, $f(0) = 1>0$ et $f(a)
<0$, donc, d'aprčs le théorčme des valeurs intermédiaires, il existe $b$
compris entre $0$ et $a$ tel que $f(b)=0$. Ce qui est impossible d'aprčs
ci-dessus. Donc $f(x)>0$ pour tout $x\in \R$.
\end{itemize}
\end{enumerate}
\end{demo}

\begin{demo}[ de l'unicité ]
Soit $f$ et $g$ qui conviennent\footnote{Début classique de la plus part des 
  démonstrations d'unicité.}. Posons $\phi = \dfrac{f}{g}$. La fonction $\phi$
est définie et dérivable sur $\R$ (car $g(x) \neq 0$ pour tout $x\in \R$,
d'aprčs le lemme). Vu que $f'=f$ et $g'=g$, il vient
\[ \phi' = \dfrac{f'g-fg'}{g^2} = \dfrac{fg-fg}{g^2} = 0 \]
Donc $\phi$ est constante et comme $\phi(0) = \dfrac{f(0)}{g(0)} = \dfrac{1}{1} =1$, pour tout
$x\in \R$ $\phi(x) = 1$. C'est-ŕ-dire $\dfrac{f}{g} = 1$, d'oů $f=g$. Il y a donc
unicité.
\end{demo}

\subsection{Propriétés de la fonction exponentielle}
\begin{prop}\label{propfdt}
\begin{enumerate}
\item La fonction $\exp$ est dérivable sur $\R$, et pour tout $x\in \R$,
  $\exp'(x) = exp(x)$.
\item $\exp(0) = 1$
\item Pour tout $x \in \R$, $\exp(-x) = \frac1{\exp(x)}$
\item Pour tout $x \in \R$, $\exp(x) >0$
\item Pour tout $x \in \R$ et tout $y \in \R$, $\exp(x+y) = \exp(x) \exp(y)$
\item Pour tout $x \in \R$ et tout $y \in \R$, $\exp(x-y) = \frac{\exp(x)}{\exp(y)}$
\item Pour tout $x \in \R$ et tout $n \in \Z$, $\exp(nx) = (\exp(x))^n$
\end{enumerate}
\end{prop}
\begin{demo}
\begin{enumerate}
\item Par définition.
\item Par définition.
\item Lemme 1, point 1.
\item Lemme 1, point 2: pour tout $x\in \R$, $\exp(x) \exp(-x) =
  1$ donc $\exp(-x) = \frac1{\exp(x)}$. 
\item On procčde comme lors de la démonstration du lemme \ref{unicite}, point
  2. Soit $y\in \R$ fixé.

La fonction définie par $\phi_y(x) = \exp(x+y) \exp(-x)$ est définie et
dérivable pour tout $x\in \R$. De plus
\begin{align*}
\phi_y'(x) &= \exp'(x+y)\exp(-x) + \exp(x+y)\exp'(-x)\times (-1)\\  
&=\exp(x+y)\exp(-x) - \exp(x+y) \exp(-x) = 0
 \end{align*}
 Ainsi $\phi_y$ est constante. Or $\phi_y(0) = \exp(0+y)\exp(-0) = \exp(y)$. 
 
 Donc,
 pour tout $x \in \R$, 
\[ \exp(x+y) \exp(-x) = \phi_y(x) = \phi_y(0) = \exp(y)\]
Par suite, d'aprčs le point 3, pour tout $x \in \R$, $\exp(x+y) = \frac1{\exp(-x)}
\exp(y) = \exp(x) \exp(y)$

\item Conséquence des points 5 et 4.
\item Récurrence.
\end{enumerate}
\end{demo}

\subsection{Le nombre $e$ et la notation puissance}

Les résultats précédents montrent que la fonction $\exp$ possčde les
propriétés algébriques des puissances.

\begin{defn}
On note $e$ le nombre $\exp(1)$ :
\[ e = \exp(1) \]
\end{defn}
\begin{rem}
$e \simeq 2{,}718 $. On peut d'ailleurs obtenir cette approximation
avec la méthode d'Euler.
\end{rem}

Recherche d'une notation simplifiée : Pour tout $n \in \N$,
\begin{eqnarray*}
  \exp (n ) &=&\exp(n\times 1) = (\exp(1))^n = e^n \hspace{4cm} \text{(d'aprčs
    la propriété \ref{propfdt}.7)}\\
  \exp (-n) &=&\frac1{\exp(n)} = \frac1{e^n} = e^{-n}\hspace{5.63cm} \text{(d'aprčs
    la propriété \ref{propfdt}.7)}\\
\end{eqnarray*}

\begin{rem}
Pour tout $x\in \R$ on note $\exp(x) =e^x$.
\end{rem}

\subsection{Résolution d'équations et d'inéquations}

\begin{prop}\label{resolution}
Pour tout réel $a$ et $b$,
\[ 
\begin{array}{rcl}
e^a=e^b&\Longleftrightarrow&a=b\\
e^a>e^b&\Longleftrightarrow&a>b\\
\end{array}
\]
\end{prop}

\begin{demo}
La fonction $\exp$ est strictement croissante.
\end{demo}

\subsection{Limites en l'infini}\label{limites}
\begin{prop}\label{Lim}
\begin{enumerate}
\item $\lim_{x \to +\infty} e^x= + \infty$.
\item $\lim_{x \to -\infty} e^x = 0$.
\end{enumerate}
\end{prop}
\begin{demo}
\begin{enumerate}
\item D'aprčs ci-dessus, pour tout $x \in \R$, $e^x \geq x+1$. Or $\lim_{x \to
    + \infty} x+1 = +\infty$. 

Donc, par comparaison, 
$\lim_{x \to +\infty} e^x= + \infty$.
\item La propriété précédente entraîne $\lim_{x \to +\infty} \frac1{e^x}
  =0$. 

De plus $\frac1{e^x} = e^{-x}$ et $\lim_{x \to -\infty} -x =
  +\infty$. Donc, par composition des limites,   
$\lim_{x \to -\infty} e^x = 0$.
\end{enumerate}
\end{demo}

\section{Applications}
\subsection{Fonctions composées du type $e^u$}
C'est un cas particulier de fonction composée ŕ connaître {\em
  impérativement}.

\begin{prop}\label{DerExpU}
Soit $u$ une fonction définie et dérivable sur un intervalle $I$. Alors
\[\left(e^u\right)' = u' e^u\]
\end{prop}
\begin{demo}
Posons $v=\exp$. Puisque $v'=\exp$, la formule de dérivation des fonctions
composées s'écrit 
$f'= (v \circ u)' = u' \times (v'\circ u) = u'e^u$.
\end{demo}

\begin{exemple}
Étudier la fonction définie par $f(x) = e^{-x}$ pour tout $x\in \R$.
\end{exemple}
La fonction $u$ définie sur $\R$ par $u(x) = -x$ a pour dérivée $u'(x) =
-1$. Pour tout $x \in \R$, $f(x)= \exp(u(x))$, donc, d'aprčs la propriété
\ref{DerExpU}, $f'(x) = u'(x) \exp(u(x)) = -1 \times \exp(-x) = -e^{-x}$.

\subsection{Croissance comparée}
\begin{prop}\label{CroissComp} ~
\vspace{-.5cm}
\begin{enumerate}
\item $\lim_{x \to +\infty} \dfrac{e^x}{x}= + \infty$.
\item $\lim_{x \to -\infty} x e^x = 0$.
\end{enumerate}
\end{prop}


\begin{demo} La démonstration est du męme type que celle de la propriété
  \ref{Lim}. 
\begin{enumerate}
\item Pour calculer la limite, nous allons minorer $e^x/x$ (pour
  $x>0$) par une fonction qui tend vers $+\infty$ en $+\infty$. 
Plus précisément, montrons que, pour tout $x >0$,
\[ \frac{x}2 \leq \frac{e^x}{x} \]
Le « $1/2$ » est décoratif, pour simplifier les calculs (voir plus
bas). Simplifions cette inégalité.
\[ \frac{x}2 \leq \frac{e^x}{x} 
\Longleftrightarrow
x^2 \leq 2 e^x 
\Longleftrightarrow
0 \leq 2 e^x -x^2
\qquad \qquad \text{($x$ est positif ici !)}\]



Pour tout $x \geq 0$, posons $\phi(x) = 2e^x -x^2$, et étudions son signe. La
fonction $\phi$ est dérivable (deux fois) sur $\R_+$ et, pour tout $x\geq 0$,
\[ \phi'(x) = 2e^x -2x \qquad \text{et} \qquad \phi''(x) = 2e^x - 2  =
2(e^x-1) \]
D'aprčs la proposition \ref{resolution}, $\phi''(x) > 0$ pour tout $x>0$ (d'oů
l'intéręt du « $1/2$ »). 

Donc $\phi'$ est strictement croissante et $\phi'(x) > \phi'(0)=2>0$ pour tout $x >0$.

Ainsi $\phi$ est elle aussi strictement croissante et $\phi(x) >
\phi(0)=2>0$. On en déduit que pour tout $x>0$, 
\[ \frac{x}2 \leq \frac{e^x}{x} \]
Or $\lim_{x \to +\infty} x/2 = +\infty$, donc, par comparaison, $\lim_{x \to
  +\infty} \dfrac{e^x}{x} = +\infty$.

En conclusion, $\lim_{x \to +\infty} \dfrac{e^x}{x} = + \infty$.
\item 
$\lim_{x \to - \infty} x e^x = \lim_{x \to +\infty} -xe^{-x}$. 
Or, pour tout $x>0$,
\[ -xe^{-x} = -\frac{x}{e^x} = - \frac1{\frac{e^x}x}
\]
Et $\lim_{x \to +\infty} \dfrac{e^x}{x} = + \infty$, donc $\lim_{x \to -\infty} x e^x = 0$.
\end{enumerate}
\end{demo}

  
Un corollaire de cette propriété est la généralisation suivante:

\begin{prop} Pour tout $n\in \N$,
%\vspace{-.5cm}
\begin{enumerate}
\item $\lim_{x \to +\infty} \dfrac{e^x}{x^n}= + \infty$.
\item $\lim_{x \to -\infty} x^n e^x = 0$.
\end{enumerate}
\end{prop}

\begin{demo}
On suppose $n \geq 1$, sinon le résultat découle immédiatement de la propriété
\ref{Lim}. Montrons le {\bf 1)}, le second point se montrant de la męme façon
qu'en \ref{CroissComp}.

Pour tout $x\in ]0;+\infty[$, on définie les fonction suivantes:
\[ f(x) = \frac{e^x}{x^n} \qquad \qquad \text{et}  \qquad \qquad
g(x) = \frac{e^x}{x}\]
Or, pour tout $x>0$,  $f(x)= \left(\dfrac{e^{x/n}}{x}\right)^n=
\left(\dfrac{e^{x/n}}{x/n}\right)^n\times \left(\dfrac1n\right)^n =C_n (g(x/n))^n$,
oů $C_n>0$ est une constante fixée (puisque $n$ est fixé). De plus, d'aprčs les
limites usuelles et la proposition \ref{CroissComp},
\[ \lim_{x \to + \infty} \dfrac{x}{n} = +\infty \qquad\qquad  \lim_{x \to +\infty} g(x) =
+\infty \qquad\qquad  \lim_{x \to + \infty} x^n = + \infty \]
Donc, par composition des limites $\lim_{x \to +\infty} \frac{e^x}{x^n}=
\lim_{x \to +\infty} f(x) = + \infty$
\end{demo}
