\section{Cosinus, sinus et tangente d'un angle aigu}

\subsection{Définitions}

\begin{defn}
Dans un triangle rectangle, le sinus d'un angle aigu est le quotient de la longueur du côté opposé à cet angle par la longueur de l'hypoténuse.
\end{defn}

\begin{defn}
Dans un triangle rectangle, le cosinus d'un angle aigu est le quotient de la longueur du côté adjacent à cet angle par la longueur de l'hypoténuse.
\end{defn}

\begin{defn}
Dans un triangle rectangle, la tangente d'un angle aigu est le quotient de la longueur du côté opposé à cet angle par la longueur du côté adjacent à cet angle.
\end{defn}

\subsection{Applications}

\begin{meth}[Calculer une longueur]

On considère un triangle $LEO$ rectangle en $E$ tel que :
$LO=5,4$cm et $\widehat{ELO}=62$\degre

\begin{enumerate}
\item Calculons la longueur du côté $[OE]$ arrondie au millimètre.

\noindent\grandscarreaux{\linewidth}{4.8cm}
\item Puis, calculons la longueur du côté $[EL]$ arrondie au millimètre.

\noindent\grandscarreaux{\linewidth}{4.8cm}
\end{enumerate}
\end{meth}

\begin{meth}[Calculer un angle]

Soit $FUN$ un triangle rectangle en $U$ tel que : $UN=8,2$cm et $UF=5,5$cm.

\medskip
\noindent Calculons la mesure de l'angle $\widehat{UNF}$ arrondie au degré.

\noindent\grandscarreaux{\linewidth}{4cm}
\end{meth}

\section{Relations trigonométriques}

\begin{prop}[Relation fondamentale]
Pour tout angle aigu $\widehat{A}$,
$$\left( \cos \widehat{A} \right)^2 + \left( \sin \widehat{A} \right)^2 = 1$$
\end{prop}

\begin{prop}
Pour tout angle aigu $\widehat{A}$,
$$\tan \widehat{A} = \dfrac{\cos \widehat{A}}{\sin \widehat{A}}$$
\end{prop}

\begin{rem}
La première formule peut aussi s'écrire $$\cos ^2 \widehat{A} + \sin ^2 \widehat{A} = 1$$
\end{rem}

\begin{ex}
\begin{enumerate}
\item Calcule la valeur exacte de $\sin \widehat{A}$ sachant que $\widehat{A}$ est un angle aigu tel que $\cos \widehat{A} = 0,8$

\noindent\grandscarreaux{\linewidth}{4cm}
\item Puis calcule la valeur exacte de $\tan \widehat{A}$

\noindent\grandscarreaux{\linewidth}{4cm}
\end{enumerate}
\end{ex}
