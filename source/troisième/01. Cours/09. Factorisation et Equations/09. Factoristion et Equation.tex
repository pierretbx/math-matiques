\chapter{Factorisation et Équations}

\section{Factoriser une expression polynomiale}

\begin{defn}
Factoriser c'est écrire une expression algébrique sous forme d'un produit.
\end{defn}

\noindent  La factorisation est utile dans les situations suivantes:
\begin{enumerate}
\item[•] simplifier un quotient,\hspace{5cm} $\bullet$ résoudre des équations.
\end{enumerate}
	
\medskip
\noindent La factorisation n'est pas évidente mais souvent astucieuse contrairement au développement. Il s'agit de reconnaître dans l'expression algébrique les formules classiques de la proposition suivante.

\begin{prop}[Boite à outils de la factorisation]
Quelque soient les nombres $a$, $b$ et $c$ réels
\begin{enumerate}
\item $ab+ac=a(b+c)$ \hspace{2cm} Facteur commun.
\item $a^2-b^2=(a-b)(a+b)$ \hspace{1.3cm} Identité remarquable.
\end{enumerate}
\end{prop}

\begin{meth}
Il n'y a pas d'algorithme pour factoriser une expression, mais vous pourrez essayer dans l'ordre:

\begin{enumerate}
\item de chercher un facteur commun:
\begin{enumerate}
\item[•] Identifier les termes de la somme (ou différence).
\item[•] Écrire chacun des termes sous forme d'un produit.
\item[•] Identifier un facteur commun à tous les termes.
\end{enumerate}

\begin{ex}
$4x+16 \quad = \quad \textcolor{orange}{4} \times x + \textcolor{orange}{4} \times 4 \quad = \quad \textcolor{orange}{4} \times (x+4)$
\end{ex}

\item  d'utiliser une identité remarquable,

\begin{ex}
$81x^2 - 25 \quad = \quad (\textcolor{orange}{9x})^2 - \textcolor{orange}{5}^2 \quad = \quad (9x-5)(9x+5)$
\end{ex}

\item de factoriser une partie de l'expression pour faire apparaître un facteur commun ou une identité remarquable,
\item et enfin de développer en espérant pouvoir ensuite factoriser.
\end{enumerate}
\end{meth}

\section{Résolution algébrique d'une équation}

\subsection{Généralités sur les équations}

\begin{defn}
Une équation est une égalité dans laquelle se trouvent des lettres (des variables) qui représentent des nombres (expression littérale).

\noindent Une équation est soit vraie, soit fausse, suivant les valeurs choisies pour remplacer les lettres.

\noindent Résoudre une équation c'est trouver toutes les valeurs qui rendent l'équation vraie.
\end{defn}

\medskip
Il n'y a pas une unique méthode de résolution des équations. Il y a presque autant de méthodes que d'équations.

\noindent Nous apprendrons certaines de ces méthodes.

\begin{prop}
On ne modifie pas les solutions d'une équation en additionnant, en soustrayant, en multipliant, ou en divisant par un même nombre non nul des deux côtés de l'égalité.
\end{prop}

\subsection{Équation polynomiale de degré 1}

\begin{defn}
Une équation polynomiale de degré $1$ (ou équation linéaire) est une équation de la forme \[ ax+b=0 \]
avec $a$ et $b$ des nombres fixés et $x$ un nombre variable.
\end{defn}

\medskip
\begin{meth}
\begin{enumerate}
\item Pour identifier l'équation du premier degré on regroupe tous les termes d'un même côté de l'égalité. Autrement dit il faut se ramener à une égalité à $0$.
\item La résolution des équations du premier degré consiste à \og isoler le $x$ \fg{}.
	
Supposons que $a$ n'est pas nul.
\begin{align*}
ax+b&=0\\
\intertext{équivaut successivement à}
ax+b {\color{orange}-b} &= 0{\color{orange}-b}\\
ax &=-b\\
\frac{ax}{{\color{orange}a}} &= \frac{-b}{{\color{orange}a}} \text{, car $a\ne 0$}\\
x &= -\frac{b}{a}
\end{align*}

\item La conclusion : L'ensemble des solutions de l'équation $ax+b=0$ est $\mathcal{S} = \left\{ -\frac{b}{a} \right\}$.
\end{enumerate}
\end{meth}

\begin{ex}
Résolvons dans $\R$ l'équation $E : 2x-13=-3x+2$.

On se ramène à un égalité à 0:
\[
\begin{aligned}
2x-13&=-3x+2\\
2x-13 {\color{orange} + 3x -2}&=-3x+2 {\color{orange} + 3x -2}\\
5x-15 &=0\\
\end{aligned}
\]
\medskip
On \og isole le $x$ \fg{} ;
\[
\begin{aligned}
5x-15 &=0\\
5x-15 {\color{orange}+15} &= 0{\color{orange}+15}\\
5x &=15\\
\frac{5x}{{\color{orange}5}} &= \frac{15}{{\color{orange}5}} \text{, car $5 \neq 0$}\\
x &= \frac{15}{5}\\
x &= 3
\end{aligned}
\]

\medskip
La conclusion : L'ensemble des solutions de l'équation $2x-13=-3x+2$ est : $\mathcal{S} = \left\{ 3 \right\}$.
\end{ex}

\section{Équation produit}

Lorsque l'on a affaire à un produit de plusieurs facteurs qui doit être égal à $0$, on utilise le théorème suivant :

\begin{prop}
Un produit de facteurs est nul si et seulement si l'un des facteurs est nul. Autrement dit :
   $$f(x)\times g(x)=0 \quad \Longleftrightarrow \quad f(x)=0 \quad \text{ et } \quad g(x)=0.$$
\end{prop}

\begin{meth}[Résoudre une équation à produit nul]\label{Eq_pr_nu}
Pour résoudre dans $\R$ l'équation $(-2x+1)(3x-6)=0$, la méthode est la suivante :
\begin{enumerate}
\item On vérifie que c'est bien une équation produit et que l'on est égal à zéro.
\item On applique la propriété ci-dessus, donc :
\[ -2x+1=0 \quad \text{ et }\quad  3x-6=0  \]

\item On résout les deux équations obtenues précédemment :
$$-2x+1=0 \quad \text{ et }\quad  3x-6=0  \quad \Longleftrightarrow \quad -2x=-1 \quad \text{ et }\quad  3x=6 \quad \Longleftrightarrow \quad x=\frac{-1}{-2}\quad \text{ et }\quad  x=2$$

\item On conclut : 
$$\mathcal{S}=\left\{\dfrac{1}{2} ; 2\right\}$$
\end{enumerate}
\end{meth}

\begin{ex}
Résolvons dans $\R$ l'équation $(x+1)(2x+4)-(x-7)(x+1)=0$\\

\renewcommand{\arraystretch}{1.5}
\begin{tabular}{p{0.1cm}l!{$\Longleftrightarrow$}l}
& $(x+1)(2x+4)-(x-7)(x+1)=0$ & $(x+1)[(2x+4)-(x-7)]=0$ \\
& & $(x+1)(x+11)=0$ \\
& & $x+1=0$ \quad et \quad $x+11=0$ \\
& & $x=-1$ \quad et \quad $x=-11$ \\
& & $\mathcal{S}=\{-11;-1\}$. \\
\end{tabular}
\end{ex}

\begin{prop}
Soit $a$ un nombre réel. L'équation $x^2=a$ possède :
   \begin{enumerate}
   \item[•] deux solutions si $a>0$ : \quad $\mathcal{S}=\{-\sqrt{a};\sqrt{a}\}$ ,
   \item[•] une solution si $a=0$ : \quad $\mathcal{S}=\{0\}$ ,
   \item[•] aucune solution si $a>0$ : \quad $\mathcal{S}=\varnothing$. 
   \end{enumerate}
\end{prop}
