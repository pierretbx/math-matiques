\section{Rappels sur les fractions}
 
\subsection{Multiplication de deux ou plusieurs fractions}
 
\begin{memento}
Pour multiplier deux fractions, on multiplie les numérateurs entre eux et on multiplie 
les dénominateurs entre eux. 
\end{memento}

\begin{ex}
Calculer et simplifier les produits suivants : \quad $\dfrac{3}{4}\times \dfrac{7}{-9}$ \quad et \quad $\dfrac{21}{-12}\times \left( -\dfrac{36}{35} \right) $
 
\medskip\noindent\grandscarreaux{\linewidth}{5.4cm}
\end{ex}

\subsection{Addition et soustraction de fractions}
 
\begin{memento}
\begin{enumerate}
\item[•] On réduit les fractions au même dénominateur (si ce n'est pas déjà)
\item[•] On ajoute ou on soustrait les numérateurs obtenus
\item[•] Enfin, on simplifie la fraction si c'est possible.
\end{enumerate}
\end{memento}

\begin{ex}
Calculer et simplifier les sommes suivantes : \quad $\dfrac{-7}{5}+ \dfrac{4}{15}$ \quad ; \quad $\dfrac{9}{5}-\dfrac{4}{3}$ \quad et \quad $4- \dfrac{2}{9}$
 
\medskip\noindent\grandscarreaux{\linewidth}{5.4cm}
\end{ex}
 
\subsection{Division de fractions}

\begin{memento}
Pour diviser par une fraction, on multiplie par la fraction inverse. 
 \end{memento}

\begin{ex}
Calculer et simplifier les quotients suivants : \quad $\dfrac{8}{9} \div \dfrac{-4}{7}$ \quad ; \quad $\dfrac{\frac{1}{2}}{\frac{-6}{7}}$ \quad et \quad $\dfrac{-8}{5} \div 4$
 
\medskip\noindent\grandscarreaux{\linewidth}{5.6cm}
\end{ex}
 
\section{Puissance d'un nombre}

L'utilisation des puissances simplifie l'écriture des produits comportant le même facteur.

\subsection{Définition}

\begin{defn}[Puissance à exposant positif]
Si $x$ est un nombre et $n$ un entier naturel, alors $x^n$, qui se lit \og $x$ puissance $n$ \fg{} ou \og $x$ exposant $n$ \fg{}, est le nombre \[ x^n=\underset{n\ \text{fois} }{\underbrace{x\times \dots \times x}} \]
Par convention : pour tout $n \in \N$, $0^{n} = 0$; pour tout $n \in \Z$, $a^1 = a$ \quad et \quad $a^0 = 1$;
\end{defn}

\begin{defn}[Puissance à exposant négatif]
Soit $n$ un nombre strictement positif et a un nombre relatif non nul. On appelle « $a$ puissance moins $n$ » le nombre noté $a^{-n}$ tel que :  $$a^{-n} = \dfrac{1}{a^n}$$
\end{defn}

\end{ex}

\subsection{Propriété}

\begin{prop}[Produit et quotient de puissances]
Soit $a$ un nombre relatif non nul; $m$ et $n$ deux entiers.
\begin{center}
$a^n \times a^m = a^{n+m}$ \hspace{0.5cm} et \hspace{0.5cm} $\dfrac{a^n}{a^m} = a^{n-m}$
\end{center}
\end{prop}

\begin{ex}
 
\medskip\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex}

\begin{prop}[Puissance de produit et de quotient]
Soit $a$ et $b$ deux nombres relatifs non nuls, soit $n$ un entier.
\begin{center}
$a^n \times b^n = (a\times b)^{n}$ \hspace{0.5cm} et \hspace{0.5cm} $\dfrac{a^n}{b^n} = \left( \dfrac{a}{b} \right)^{n}$
\end{center}
\end{prop}

\begin{prop}[Puissance de puissance]
Soit $a$ un nombre relatif ; soit $m$ et $n$ des entiers
\begin{center}
$\left( a^n \right)^{m} = a^{n \times m}$
\end{center}
\end{prop}

\end{ex}

\subsection{L'écriture scientifique}

\begin{prop}
Tout nombre non nul peut s'écrire sous la forme : $a \times 10^n$, où $a$ est un nombre décimal écrit avec un seul chiffre avant la virgule autre que $0$, et $n$ est un nombre entier relatif. 

\noindent On appelle cette notation la notation scientifique, elle est unique pour un nombre donné.
\end{prop}

\begin{ex}
Donnons l'écriture scientifique des nombres suivants : \quad $38$ \quad ; \quad $0,0562$ \quad et \quad $-97631$ .

\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
