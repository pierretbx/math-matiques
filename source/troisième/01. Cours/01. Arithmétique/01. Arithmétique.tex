\section{Division euclidienne}

\begin{defn}[Division euclidienne]
Soient $a$ et $b$ deux nombres entiers avec $b>0$.
Effectuer la division euclidienne de $a$ par $b$, c'est trouver deux nombres entiers $q$ et $r$ tels que :
\[
a=b\times q+r \quad \text{et} \quad 0\leq r < b.
\]
Le nombre $q$ est le quotient et le nombre $r$ est le reste.
\end{defn}

\begin{ex}
Dans la division euclidienne de $157$ par $12$, on a :
\[
157=12\times 13+1.
\]
Le quotient est $13$ et le reste est $1$.
\end{ex}

\begin{ex}
Effectuer les divisions euclidiennes suivantes.

\begin{multicols}{2}
\begin{enumerate}
\item $238$ par $7$.
\item $654$ par $15$.
\item $1000$ par $24$.
\item $2026$ par $9$.
\end{enumerate}
\end{multicols}

\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}

\section{Diviseurs et multiples}

\begin{defn}[Diviseur et multiple]
Soient $a$ et $b$ deux nombres entiers avec $b\neq 0$.
On dit que $b$ est un diviseur de $a$ lorsque le reste de la division euclidienne de $a$ par $b$ est égal à $0$.

Dans ce cas, il existe un entier $k$ tel que :
\[
a=b\times k.
\]
On dit aussi que $a$ est un multiple de $b$ ou que $a$ est divisible par $b$.
\end{defn}

\begin{ex}
$84=7\times 12$, donc $7$ est un diviseur de $84$ et $84$ est un multiple de $7$.

\medskip
$84=3\times 28$, donc $3$ et $28$ sont aussi des diviseurs de $84$.
\end{ex}

\begin{prop}[Critères de divisibilité]
Un nombre entier est divisible :
\begin{itemize}
\item par $2$ lorsque son chiffre des unités est $0$, $2$, $4$, $6$ ou $8$ ;
\item par $3$ lorsque la somme de ses chiffres est divisible par $3$ ;
\item par $5$ lorsque son chiffre des unités est $0$ ou $5$ ;
\item par $9$ lorsque la somme de ses chiffres est divisible par $9$ ;
\item par $10$ lorsque son chiffre des unités est $0$.
\end{itemize}
\end{prop}

\begin{ex}
Le nombre $4\,185$ est divisible par $3$ car $4+1+8+5=18$ et $18$ est divisible par $3$.
Il est aussi divisible par $5$ car son chiffre des unités est $5$.
\end{ex}

\section{Nombres premiers}

\begin{defn}[Nombre premier]
Un nombre premier est un nombre entier positif qui possède exactement deux diviseurs : $1$ et lui-même.
\end{defn}

\begin{ex}
$2$, $3$, $5$, $7$, $11$, $13$, $17$ et $19$ sont des nombres premiers.

\medskip
$1$ n'est pas un nombre premier car il ne possède qu'un seul diviseur.

\medskip
$21$ n'est pas un nombre premier car il possède plus de deux diviseurs : $1$, $3$, $7$ et $21$.
\end{ex}

\begin{thm}[Décomposition en produit de facteurs premiers]
Tout nombre entier supérieur ou égal à $2$ peut se décomposer en produit de facteurs premiers.
Cette décomposition est unique, à l'ordre des facteurs près.
\end{thm}

\begin{demo}
L'existence et l'unicité de cette décomposition sont admises en classe de troisième.
L'idée importante est que les nombres premiers jouent le rôle de \og briques élémentaires \fg{} pour les nombres entiers.
\end{demo}

\begin{meth}[Décomposer un nombre en produit de facteurs premiers]
On divise le nombre par les nombres premiers dans l'ordre croissant : $2$, $3$, $5$, $7$, $11$, etc.
On recommence tant que c'est possible, jusqu'à obtenir $1$.
\end{meth}

\begin{ex}
Décomposons $360$ en produit de facteurs premiers.
\[
360=2\times 180
=2\times 2\times 90
=2^3\times 3^2\times 5.
\]
Ainsi :
\[
360=2^3\times 3^2\times 5.
\]
\end{ex}

\begin{ex}
Décomposer les nombres suivants en produit de facteurs premiers.

\begin{multicols}{3}
\begin{enumerate}
\item $84$
\item $126$
\item $450$
\end{enumerate}
\end{multicols}

\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}

\section{Fractions irréductibles}

\begin{defn}[Fraction irréductible]
Une fraction est irréductible lorsque son numérateur et son dénominateur n'ont pas de diviseur commun autre que $1$.
\end{defn}

\begin{meth}[Rendre une fraction irréductible]
Pour rendre une fraction irréductible, on peut :
\begin{enumerate}
\item décomposer le numérateur et le dénominateur en produit de facteurs premiers ;
\item simplifier tous les facteurs communs ;
\item écrire la fraction obtenue.
\end{enumerate}
\end{meth}

\begin{ex}
Rendons la fraction $\dfrac{126}{210}$ irréductible.
\[
126=2\times 3^2\times 7
\quad \text{et} \quad
210=2\times 3\times 5\times 7.
\]
Donc :
\[
\dfrac{126}{210}
=\dfrac{2\times 3^2\times 7}{2\times 3\times 5\times 7}
=\dfrac{3}{5}.
\]
\end{ex}

\section{Recherche de diviseurs communs}

\begin{meth}[Déterminer des diviseurs communs]
Pour déterminer les diviseurs communs de deux nombres, on peut décomposer les deux nombres en produit de facteurs premiers, puis repérer les facteurs communs.
\end{meth}

\begin{ex}
On cherche les diviseurs communs de $72$ et $120$.
\[
72=2^3\times 3^2
\quad \text{et} \quad
120=2^3\times 3\times 5.
\]
Les facteurs communs sont $2^3$ et $3$.
Le plus grand diviseur commun est donc :
\[
2^3\times 3=24.
\]
\end{ex}

\begin{ex}
Un fleuriste dispose de $84$ roses et de $126$ tulipes.
Il veut préparer le plus grand nombre possible de bouquets identiques, sans reste.

\begin{enumerate}
\item Décomposer $84$ et $126$ en produit de facteurs premiers.
\item Déterminer le nombre maximal de bouquets.
\item Donner la composition de chaque bouquet.
\end{enumerate}

\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
