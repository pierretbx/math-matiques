\begin{minipage}{9cm}

\begin{minipage}{7cm}
\begin{center}
   \renewcommand{\arraystretch}{1.5}   
   \begin{tabular}{|c|*{4}{C{0.4}|}c|}
      \hline       
      \cellcolor{dftcolor!20}{Valeurs} & $x_1$ & $x_2$ & $\dots$ & $x_p$ & \cellcolor{dftcolor!20}{Total} \\
      \hline
      \cellcolor{dftcolor!20}{Effectifs} & $n_1$ & $n_2$ & $\dots$ & $n_p$ & $N$ \\ 
      \hline
      %Frﾃｩq. en \% & $0$ & $3$ & $7$ & $3$ & $3$ & $7$ & $10$ & $17$ & $20$ & $7$ & $10$ & $0$ & $7$ & $0$ & $3$ & $3$ & $0$ & $0$ & $0$ \\
