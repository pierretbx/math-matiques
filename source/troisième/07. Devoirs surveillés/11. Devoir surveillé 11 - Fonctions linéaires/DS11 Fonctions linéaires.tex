\section*{Devoir surveillé 11 - Fonctions linéaires}
\addcontentsline{toc}{section}{Devoir surveillé 11 - Fonctions linéaires}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Reconnaître]
Parmi les fonctions $f(x)=3x$, $g(x)=x+4$, $h(x)=-2x$ et $k(x)=x^2$, indiquer celles qui sont linéaires.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Calculer]
Soit $f(x)=-4x$. Calculer $f(3)$, $f(-2)$ et l'antécédent de $20$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Représenter]
Tracer la représentation graphique de la fonction $x\mapsto 2x$.
\medskip\noindent\grandscarreaux{\linewidth}{7cm}
\end{ex}
