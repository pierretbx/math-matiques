\section*{Devoir surveillé 2 - Théorème de Thalès}
\addcontentsline{toc}{section}{Devoir surveillé 2 - Théorème de Thalès}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Configuration de Thalès]
Dans le triangle $ABC$, les points $M$ et $N$ appartiennent respectivement à $[AB]$ et $[AC]$, et $(MN)\parallel(BC)$. On donne $AM=4$, $AB=10$ et $AC=15$. Calculer $AN$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Réciproque]
Dans le triangle $RST$, $A\in[RS]$ et $B\in[RT]$. On donne $RA=6$, $RS=10$, $RB=9$ et $RT=15$. Les droites $(AB)$ et $(ST)$ sont-elles parallèles ? Justifier.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
\begin{ex}[Agrandissement]
Un triangle est réduit avec un coefficient $0,6$. Un côté du grand triangle mesure $18$ cm. Calculer la longueur correspondante du petit triangle.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
