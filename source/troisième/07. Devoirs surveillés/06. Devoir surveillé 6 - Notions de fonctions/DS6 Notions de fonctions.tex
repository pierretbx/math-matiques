\section*{Devoir surveillé 6 - Notions de fonctions}
\addcontentsline{toc}{section}{Devoir surveillé 6 - Notions de fonctions}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Image et antécédent]
On donne $f(x)=2x-5$. Calculer l'image de $4$ et chercher un antécédent de $7$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Tableau]
Compléter le tableau de valeurs de $g(x)=x^2-3$ pour $x=-2$, $0$, $1$ et $3$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Graphique]
Lire sur un graphique fourni par le professeur l'image de $2$ et les antécédents de $0$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
