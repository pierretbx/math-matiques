\section*{Devoir surveillé 5 - Trigonométrie}
\addcontentsline{toc}{section}{Devoir surveillé 5 - Trigonométrie}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Choisir le bon rapport]
Dans un triangle $ABC$ rectangle en $A$, on connaît $AB=7$ cm, $AC=9$ cm et l'angle $\widehat{ABC}$. Indiquer quels côtés sont adjacent, opposé et hypoténuse par rapport à cet angle.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Calculer une longueur]
Dans un triangle rectangle, l'hypoténuse mesure $12$ cm et un angle aigu mesure $35^\circ$. Calculer le côté opposé à cet angle. Arrondir au millimètre.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Calculer un angle]
Dans un triangle rectangle, le côté opposé à un angle aigu mesure $5$ cm et l'hypoténuse $13$ cm. Calculer la mesure de cet angle, arrondie au degré.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
