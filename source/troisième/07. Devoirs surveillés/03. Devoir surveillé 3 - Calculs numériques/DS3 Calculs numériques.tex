\section*{Devoir surveillé 3 - Calculs numériques}
\addcontentsline{toc}{section}{Devoir surveillé 3 - Calculs numériques}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Calculs]
Calculer : $(-7)+(+12)$, $(-6)\times 9$, $(-48)\div 6$ et $2^3\times2^4$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Priorités]
Calculer en détaillant : $A=-5+8\times(-3)$ et $B=(7-12)^2-4\times3$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Notation scientifique]
Écrire en notation scientifique : $5400000$, $0,00072$ et $-31200$.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
