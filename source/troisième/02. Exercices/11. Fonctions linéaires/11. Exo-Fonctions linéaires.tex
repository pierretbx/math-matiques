\section*{Fonctions linéaires}

\begin{ex}
Dire si les fonctions suivantes sont linéaires.
\begin{multicols}{2}
\begin{enumerate}
\item $f(x)=3x$
\item $g(x)=x+2$
\item $h(x)=-0,5x$
\item $k(x)=x^2$
\end{enumerate}
\end{multicols}
\end{ex}

\begin{ex}
Pour $f(x)=-4x$, calculer $f(0)$, $f(3)$, $f(-2)$ et l'antécédent de $20$.
\end{ex}

\begin{ex}
Une situation est proportionnelle : $8$ kg coûtent $19,20$ euros.
\begin{enumerate}
\item Déterminer le coefficient de proportionnalité.
\item Définir la fonction linéaire donnant le prix en fonction de la masse.
\item Calculer le prix de $12$ kg.
\end{enumerate}
\end{ex}
