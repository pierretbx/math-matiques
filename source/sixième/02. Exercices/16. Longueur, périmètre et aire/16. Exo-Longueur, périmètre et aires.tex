\begin{ex}

\begin{enumerate}
\item Un triangle $GTU$, isocèle en $G$, tel que $GU = 30$mm et $TU=4$cm.
\item Un triangle équilatéral $RTF$ de coté $40$mm.
\end{enumerate}
\end{ex}

\begin{ex}
