\subsection*{Droite, demi-droite et segment}
\addcontentsline{toc}{subsection}{Droite, demi-droite et segment}

\begin{ex}
Place trois points non alignés $R$, $I$ et $Z$. Trace :
\begin{enumerate}
\begin{tabular}{m{5cm}m{5cm}m{5cm}}
\item en rouge, la droite $(RI)$ ;&
\item en bleu, le segment $(RZ)$ ;&
\item en vert, la demi-droite $(IZ)$.\\
\end{tabular}
\end{enumerate}
\end{ex}

\begin{ex}
Trace une droite $(d)$.
\begin{enumerate}
\item Place deux points $S$ et $A$ sur cette droite.
\item Donne deux autres façons de nommer la droite $(d)$.
\item Placer un point $C$ qui n'appartient pas à la droite $(d)$.
\item Le point $A$ appartient-il à la droite $(SC)$ ?
\end{enumerate}
\end{ex}

\subsection*{Longueur et milieu d'un segment}
\addcontentsline{toc}{subsection}{Longueur et milieu d'un segment}

\begin{ex}

\medskip
\begin{enumerate}
\item Tracer un segment $[AB$] de longueur $8,2$ cm.
\item Placer son milieu $C$. Coder la figure.
\item Quelles égalités de longueurs peut-on écrire ?
\item Quelle est la longueur du segment $[CB]$ ?
\end{enumerate}
\end{ex}

\begin{ex}

\medskip
\begin{enumerate}
\item Tracer un segment $[TP]$ de longueur $11,7$ cm. 

Placer sur ce segment le point $I$ situé à $5,8$ cm du point $T$.
\item \begin{enumerate}
\item Le point $I$ semble-t-il être au milieu du segment $[TP]$ ?
\item Le point $I$ est-il le milieu du segment $[TP]$ ? Justifier la réponse.
\end{enumerate}
\end{enumerate}
\end{ex}

\begin{ex}

\medskip
\begin{enumerate}
\item Tracer un segment $[IL]$ de longueur $8,1$ cm. 

Placer sur ce segment le point $D$ à $2,7$ cm du point $I$ et le point $O$ à $2,7$ cm du point $L$.
\item Calculer la longueur $DO$. Coder la figure.
\item Citer des milieux. Justifier chaque réponse.
\end{enumerate}
\end{ex}
