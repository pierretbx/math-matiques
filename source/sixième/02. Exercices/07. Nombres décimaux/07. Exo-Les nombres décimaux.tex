\begin{enumerate}\medskip
\item $4$ unités $6$ dixièmes= \dots ~\dots ~\dots dixièmes.\medskip

\item  \dots ~\dots ~\dots unité \dots ~\dots ~\dots dixièmes = $150$ centièmes.
\end{enumerate}
\end{ex}
