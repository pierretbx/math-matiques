\medskip
\paragraph{Exercice 2} \hfill \textbf{4 points}

\noindent Complète les phrases avec les mots : \textit{droite, segment, demi-droite, extrémité, appartient}.
\begin{enumerate}
\item $[AB]$ désigne le \dotfill d'extrémités $A$ et $B$.
\item $(AB)$ désigne la \dotfill passant par $A$ et $B$.
\item $[AB)$ désigne la \dotfill d'origine $A$ passant par $B$.
\item Si un point est sur une droite, on dit qu'il \dotfill à cette droite.
\end{enumerate}
