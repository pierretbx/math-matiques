\section*{Devoir surveillé 12 - Gestion de données}
\noindent\textit{Lis attentivement les tableaux et soigne les représentations graphiques.} \hfill \textit{Durée : 45 minutes}

\medskip
\paragraph{Exercice 1} \hfill \textbf{6 points}

\noindent Voici les sports préférés des élèves d'une classe.

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{|c|c|c|c|c|c|}
\hline
Sport & Football & Danse & Basket & Natation & Tennis\\\hline
Effectif & 8 & 5 & 6 & 4 & 2\\\hline
\end{tabular}
\end{center}

\begin{enumerate}
\item Combien d'élèves ont été interrogés ?
\item Quel est le sport le plus choisi ?
\item Combien d'élèves ont choisi un sport d'eau ou de raquette ?
\item Quelle fraction des élèves a choisi le basket ?
\end{enumerate}

\paragraph{Exercice 3} \hfill \textbf{4 points}

\noindent On a relevé les températures à midi pendant une semaine :
\[
12^\circ C,\quad 15^\circ C,\quad 14^\circ C,\quad 16^\circ C,\quad 15^\circ C,\quad 13^\circ C,\quad 17^\circ C.
\]
\begin{enumerate}
\item Quelle est la plus petite température ?
\item Quelle est la plus grande température ?
\item Calcule la température moyenne de la semaine.
\end{enumerate}

\paragraph{Exercice 4} \hfill \textbf{4 points}

\noindent Dans un diagramme circulaire, la moitié du disque représente les élèves qui viennent en bus. Un quart représente ceux qui viennent à pied. Le reste vient en voiture.
\begin{enumerate}
\item Quelle fraction du disque représente les élèves qui viennent en voiture ?
\item Complète le diagramme circulaire.
\end{enumerate}
