\begin{ex}[Fractions égales à un nombre entier]
Complète pour que chaque égalité soit vraie.

\begin{enumerate}
\begin{tabularx}{\linewidth}{*{4}{>{\arraybackslash}X}}
\item $3=\dfrac{\ldots}{5}$ &
\item $7=\dfrac{\ldots}{4}$ &
\item $6=\dfrac{\ldots}{10}$ &
\item $9=\dfrac{\ldots}{3}$
\end{tabularx}
\end{enumerate}
\end{ex}

\begin{ex}[Placer sur une demi-droite graduée]
Place les nombres suivants sur la demi-droite graduée :
\[
\frac{1}{4}\quad;\quad \frac{3}{4}\quad;\quad \frac{5}{4}\quad;\quad \frac{7}{4}.
\]

\begin{center}
\begin{tikzpicture}[x=3.2cm]
\draw[->,thick] (0,0) -- (2.25,0);
\foreach \x in {0,0.25,...,2} {
  \draw (\x,0.08) -- (\x,-0.08);
}
\foreach \x/\lab in {0/0,1/1,2/2} {
  \draw (\x,0.12) -- (\x,-0.12) node[below] {\lab};
}
\end{tikzpicture}
\end{center}
\end{ex}

\begin{ex}[Comparer des fractions]
Complète avec $<$, $>$ ou $=$.

\begin{enumerate}
\begin{tabularx}{\linewidth}{*{4}{>{\arraybackslash}X}}
\item $\dfrac{3}{8}\ \ldots\ \dfrac{5}{8}$ &
\item $\dfrac{7}{4}\ \ldots\ 1$ &
\item $\dfrac{6}{6}\ \ldots\ 1$ &
\item $\dfrac{9}{5}\ \ldots\ \dfrac{8}{5}$
\end{tabularx}
\end{enumerate}
\end{ex}

\begin{ex}[Problème]
Au goûter, on partage équitablement $5$ pizzas identiques entre $8$ élèves.
\begin{enumerate}
\item Quelle fraction de pizza reçoit chaque élève ?
\item Cette quantité est-elle inférieure ou supérieure à une demi-pizza ? Justifie.
\end{enumerate}
\end{ex}
