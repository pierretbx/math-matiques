\section*{Devoir surveillé 13 - Symétrie axiale}
\noindent\textit{Les constructions doivent être précises. Tu peux utiliser règle, équerre et compas.} \hfill \textit{Durée : 45 minutes}

\medskip
\paragraph*{Exercice 1} \hfill \textbf{4 points}

\noindent Construis les symétriques des points $A$, $B$, $C$ et $D$ par rapport à la droite $(d)$.

\begin{center}
\begin{tikzpicture}[scale=0.75,line cap=round,line join=round]
\draw[lightgray] (-5,-3) grid (5,3);
\draw[red,line width=1.4pt] (0,-3)--(0,3);
\node[red,anchor=west] at (0,2.7) {$(d)$};
\foreach \p/\x/\y in {A/-4/2,B/-2/-1,C/3/1,D/1/-2}
  {\fill[blue] (\x,\y) circle (2pt) node[above right,black] {$\p$};}
\end{tikzpicture}
\end{center}
