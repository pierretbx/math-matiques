\section*{Devoir surveillé 11 - Opérations sur les nombres décimaux}
\noindent\textit{La calculatrice est interdite. Les opérations doivent être posées lorsque c'est demandé.} \hfill \textit{Durée : 45 minutes}

\medskip
\paragraph{Exercice 1} \hfill \textbf{4 points}

\noindent Calcule mentalement.
\begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{*{4}{X}}
$4,7+3,8=$ & $12,5-4,6=$ & $6,2\times 10=$ & $58,4\div 100=$\\
$0,75+0,25=$ & $9-3,45=$ & $0,36\times 1000=$ & $420\div 10=$
\end{tabularx}
\end{center}

\medskip
\paragraph{Exercice 2} \hfill \textbf{5 points}

\noindent Pose et effectue les opérations.
\begin{enumerate}
\item $38,47+9,508+125,6$
\item $804,3-57,86$
\item $46,25-8,7$
\end{enumerate}

\medskip
\paragraph{Exercice 3} \hfill \textbf{5 points}

\noindent Pose et effectue les multiplications.
\begin{enumerate}
\item $7,4\times 3,8$
\item $12,05\times 4,6$
\end{enumerate}

\medskip\noindent\grandscarreaux{\linewidth}{6cm}

\medskip
\paragraph{Exercice 4} \hfill \textbf{6 points}

\noindent Lina achète $3$ cahiers à $2,85$ euros chacun, un compas à $4,60$ euros et une règle à $1,75$ euro.
\begin{enumerate}
\item Calcule le prix des trois cahiers.
\item Calcule le montant total de ses achats.
\item Elle paie avec un billet de $20$ euros. Combien le vendeur doit-il lui rendre ?
\end{enumerate}
