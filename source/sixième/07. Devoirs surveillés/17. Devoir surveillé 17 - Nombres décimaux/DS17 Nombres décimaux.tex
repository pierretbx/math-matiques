\medskip

\begin{ex}[Écritures décimales]
Donne l'écriture décimale de chaque nombre.

\begin{enumerate}
\begin{tabularx}{\linewidth}{*{4}{>{\arraybackslash}X}}
\item $\dfrac{47}{10}$ &
\item $\dfrac{309}{100}$ &
\item $\dfrac{8}{1000}$ &
\item $\dfrac{1205}{100}$
\end{tabularx}
\end{enumerate}
\end{ex}

\begin{ex}[Fractions décimales]
Écris chaque nombre sous la forme d'une fraction décimale.

\begin{enumerate}
\begin{tabularx}{\linewidth}{*{4}{>{\arraybackslash}X}}
\item $5,8$ &
\item $0,37$ &
\item $12,009$ &
\item $104,25$
\end{tabularx}
\end{enumerate}
\end{ex}

\begin{ex}[Décomposer]
Complète comme dans l'exemple : $23,47=23+\dfrac{4}{10}+\dfrac{7}{100}$.

\begin{enumerate}
\item $6,35=$ \dotfill
\item $72,108=$ \dotfill
\item $0,504=$ \dotfill
\end{enumerate}
\end{ex}

\begin{ex}[Demi-droite graduée]
Écris l'abscisse de chaque point.

\begin{center}
\begin{tikzpicture}[x=2.4cm]
\draw[->,thick] (0,0) -- (5.4,0);
\foreach \x in {0,1,...,5} {
  \draw (\x,0.12) -- (\x,-0.12) node[below] {\x};
}
\foreach \x in {0.1,0.2,...,5.3} {
  \draw (\x,0.06) -- (\x,-0.06);
}
\foreach \x/\lab in {1.3/A,2.7/B,4.05/C} {
  \draw[blue,thick] (\x,0.16) -- (\x,-0.16);
  \node[above,blue] at (\x,0.18) {$\lab$};
}
\end{tikzpicture}
\end{center}

$A=$ \dotfill \hfill $B=$ \dotfill \hfill $C=$ \dotfill
\end{ex}
