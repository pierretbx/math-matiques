\section{Périmètre}

\begin{defn}
Le périmètre d'une figure est la longueur que l'on parcourt lorsqu'on fait le tour de la figure.
\end{defn}

\begin{ex}
\begin{flushright}
\includegraphics[scale=0.25]{17. Longueur, périmètre et aire 1.png}\hspace{3cm}
\end{flushright}

\vspace{-1.5cm}
\noindent Le périmètre de cette figure est $16$ unités de longueur.

\end{ex}

\subsection{Périmètres des figures usuelles}

\begin{prop}[périmètre du carré]
On considère un carré de coté $c$.
\begin{center}
Périmètre = $4 \times$ longueur du côté \quad \quad \quad \quad  $\mathcal{P} = 4 \times c$
\end{center}
\end{prop}

\begin{ex}
Le périmètre d'un carré de côté $3$cm est :

\noindent\grandscarreaux{\linewidth}{1.6cm}
%\noindent$P = 4 \times 3 = 12$cm.
\end{ex}

\begin{prop}[périmètre d'un rectangle]
On considère un rectangle de longueur $L$ et de largeur $l$.
\begin{center}
Périmètre = $2 \times$ (Longueur $+$ largeur) \quad \quad \quad \quad $\mathcal{P} = 2 \times( L + l )$
\end{center}
\end{prop}

\begin{ex}
Le périmètre d'un rectangle de longueur $7$cm et de largeur $5$cm est :

%\begin{eqnarray*}

%P &=& 2 \times 12 \\
%P &=& 24 cm
%\end{eqnarray*}
\end{ex}

\begin{prop}[périmètre d'un triangle]
Le périmètre d'un triangle est égal à la somme des longueurs des trois côtés. (\textbf{exprimées dans la même unité}).
$$\mathcal{P} = AB + BC + CA$$
\end{prop}

\begin{prop}[périmètre d'un cercle]
Le périmètre (ou longueur ou circonférence) d'un cercle de rayon $R$ et de diamètre $D$ est égal à :
$$\mathcal{P} = 2 \times R \times \pi \hspace{2cm} \mathcal{P} = \pi \times D$$
\end{prop}

\begin{rem}
Le nombre $\pi$ se prononce \og pi \fg{}. Son écriture est infinie.

\noindent Les premières décimales sont:
$$\pi \approx 3,1415926535 8979323846 2643383279 5028841971 6939937510 5820974944 5923078164 0628620899 8628034825...$$

\noindent Dans la pratique, on prend: $\pi \approx 3,14$.
\end{rem}

\item Calculons la longueur d'un cercle de rayon $3$cm.

\noindent\grandscarreaux{\linewidth}{3.2cm}
\item Calculons la longueur d'un demi-cercle de diamètre $4$cm.

\noindent\grandscarreaux{\linewidth}{3.2cm}
\end{enumerate} 
\end{ex}

\section{Aire}

\begin{defn}
La surface d'une figure est la partie qui se trouve à l'intérieur de la figure.
\noindent L'aire est la mesure de la surface .
\end{defn}

\medskip
\noindent La surface du carré peut être représentée par un nombre. Ce nombre s'appelle l'aire du carré.

\begin{prop}[Aire d'un disque]
L'aire d'un disque de rayon $R$ est égal à :
$$\mathcal{A} = \pi \times R^2$$
\end{prop}
