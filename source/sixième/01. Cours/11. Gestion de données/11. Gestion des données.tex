\section{Tableaux}

\begin{prop}
Un tableau permet de regrouper des données, de lire facilement des informations.
\end{prop}

\begin{ex}
Les tableaux ci-dessous sont des tableaux à simple entrée.
\begin{center}
\renewcommand{\arraystretch}{2.5}
\begin{tabularx}{14cm}{|*{2}{>{\centering \arraybackslash}X|}}
\hline
\rowcolor{lightgray!50}Continent & Population en 2008 en million d'habitants\\\hline
Afrique & $\np{987}$\\\hline
Asie & $\np{4 075}$\\\hline
Europe & $\np{731}$\\\hline
Amérique Latine & $\np{579}$\\\hline
Amérique du Nord & $\np{342}$\\\hline
Océanie	& $\np{35}$ \\\hline
\end{tabularx}
\end{center}

\medskip
\noindent Que signifie les nombres $731$ et $35$ ?

\medskip
\noindent\grandscarreaux{\linewidth}{8cm}
\end{ex}

Un graphique cartésien permet de représenter l'évolution d'une grandeur en fonction d'une autre.

\subsection{Diagramme en bâton}

\begin{prop}
Dans un diagramme en bâton, les hauteurs des bâtons sont proportionnelles aux quantités représentées.
\end{prop}

\begin{ex}

\begin{center}
\begin{tikzpicture}[scale=1.3]
\begin{axis}[xscale=1.5,
	x tick label style={
		/pgf/number format/1000 sep=},
	ylabel=Population (en millard d'habitants),
	enlargelimits=0,
	legend style={at={(0.3,-0.1)},
	anchor=north,legend columns=-1},
	ybar interval=0.8,
]
\addplot 
	coordinates { (2008,0.987) (1995,0.700) (1982,1.2)};
\addplot 
	coordinates { (2008,4.075) (1995,3.450) (1982,1.2)};
\addplot 
	coordinates { (2008,0.720) (1995,0.720) (1982,1.2)};
\addplot 
	coordinates { (2008,0.600) (1995,0.500) (1982,1.2)};
\addplot 
	coordinates { (2008,0.350) (1995,0.300) (1982,1.2)};
\addplot 
	coordinates { (2008,0.14) (1995,0.13) (1982,0.1)};
\legend{Afrique,Asie,Europe,Amérique Latine,Amérique du Nord,Océanie}
\end{axis}
\end{tikzpicture}
\end{center}

\begin{enumerate}
\item[•] Que permet de visualiser d'un premier coup d'œil ce diagramme ?

\noindent\grandscarreaux{\linewidth}{4.8cm}
\item[•] Quel est le continent où il y a le plus d'écart de population entre $1995$ et $2008$ ?

\noindent\grandscarreaux{\linewidth}{4.8cm}
\end{enumerate}
\end{ex}


\subsection{Diagramme circulaire}

\begin{prop}
Dans un diagramme circulaire (ou semi-circulaire), les mesures des angles sont proportionnelles aux quantités représentées.
\end{prop}

\begin{ex} Ci-contre on a construit un diagramme circulaire représentant la population en $2008$, en pourcentage de la population mondiale, par continent.
