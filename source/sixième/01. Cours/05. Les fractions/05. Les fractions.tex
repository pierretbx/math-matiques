\section{Vocabulaire}

\noindent \begin{minipage}{6cm}
\huge{$$\dfrac{a}{b}$$}
\end{minipage}
\begin{minipage}{10cm}
$a$ est le \dotfill \medskip

$b$ est le \dotfill et $b$ est différent \dotfill
\end{minipage}

\begin{defn}

$\dfrac{a}{b}$ est une fraction si son numérateur $a$ et son dénominateur $b$ sont des \dotfill
\end{defn}

\begin{ex}
$\dfrac{15}{18}$ est une fraction tandis que $\dfrac{1,5}{18}$ et $\dfrac{1,5}{1,8}$ sont des nombres en écriture fractionnaire.
\end{ex}

\begin{prop}
Tout nombre entier peut s'écrire sous la forme d'une fraction.
\end{prop}

\begin{ex}
Le nombre $21$ peut s'écrire $21=\dfrac{21}{1}$. C'est aussi le cas pour tous les autres nombres entiers
\end{ex}

\section{Nombre fraction}

\begin{defn}
La fraction $\dfrac{a}{b}$ est le nombre qui, multiplié par $b$, donne $a$. Soit $\dfrac{a}{b} \times b = a$
\end{defn}

\section{Comparaison d'une fraction à 1}

\begin{prop}
\begin{enumerate}
\item[•]Si le numérateur est inférieur au dénominateur alors la fraction est\dotfill \medskip
\item[•]Si le numérateur et le dénominateur sont égaux alors la fraction est\dotfill \medskip
\item[•]Si le numérateur est supérieur au dénominateur alors la fraction est \dotfill
\end{enumerate}
\end{prop}

\begin{ex}
Compare les fractions $\dfrac{11}{15}$, $\dfrac{15}{15}$ et $\dfrac{17}{15}$ à $1$.

\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}

\section{Fraction et demi-droite graduée}

\begin{prop}
Pour placer la fraction $\dfrac{a}{b}$ sur une demi-droite graduée, on partage chaque unité en $b$ parts égales, puis on compte $a$ parts depuis $0$.
\end{prop}

\begin{ex}
Place les fractions $\dfrac{7}{4}$ et $\dfrac{11}{4}$ sur la demi-droite graduée.

\section{Encadrement entre deux nombres entiers consécutifs}

\begin{prop}
On effectue la division euclidienne du numérateur par le dénominateur. On obtient un quotient entier qui correspond à la valeur approchée à l'unité par défaut du quotient.
\end{prop}

\begin{ex}
Encadre la fraction $\dfrac{39}{7}$ entre deux entiers consécutifs.

\end{ex}
