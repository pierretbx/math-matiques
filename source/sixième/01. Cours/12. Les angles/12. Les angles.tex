\section{Notion d'angle}

\noindent\begin{minipage}{14cm}
\begin{defn}
Un angle est une proportion de plan qui est délimitée par deux demi-droites de même origine.
\end{defn}
\end{minipage}

\noindent\begin{minipage}{12cm}
\begin{prop}
L'origine de ces deux demi-droites est le sommet de l'angle.

\noindent Ces deux demi-droites sont les côtés de l'angle.
\end{prop}
\end{minipage}


\begin{ex}
\noindent\begin{minipage}{0.55\linewidth}
Le point $O$, en rouge, est le sommet de l'angle.

\medskip
\noindent Les demi-droites $[OA)$ et $[OB)$, en noir, sont les côtés de l'angle en vert.
\end{minipage}

\begin{prop}
Un angle se note à l'aide de trois lettres. La lettre entre les deux autres est celle qui désigne le sommet de l'angle
\end{prop}

\begin{ex}
\noindent\begin{minipage}{0.55\linewidth}
L'angle vert se note $\widehat{CID}$ ; $\widehat{EIF}$ ; $\widehat{CIF}$ ; $\widehat{DIC}$ ; $\widehat{DIE}$ ; $\widehat{FIC}$ ; $\widehat{FIE}$ ; \dots

\medskip
\noindent Dans toutes ces notations, la lettre $I$ désigne le sommet de l'angle.
\end{minipage}

\section{Mesure d'un angle}

\subsection{Unité : le degré}

Au collège, l'unité de mesure d'angle est le degré.

\begin{defn}
Un angle plat peut être partagé en $180$ parties égales.

\noindent Un degré (noté \degre) est la mesure de chacune de ces parties.
\end{defn}

\begin{prop}
Lorsque la mesure d'un angle $\widehat{AOB}$ est égale à 40\degre, on note : $\widehat{AOB} = 40$\degre
\end{prop}

\begin{rem}
\begin{enumerate}
\item[•] Un angle plat mesure 180 \degre
\item[•] Pour mesurer un angle, on utilise un rapporteur.
\item[•] La plupart des rapporteurs sont gradués dans les deux sens.
\end{enumerate}
\end{rem}

\begin{ex}
\begin{center}
\rapporteurAngle
\end{center}

\bigskip
La mesure de l'angle $\widehat{AOB}$ est environ égale à \dotfill
\end{ex}
