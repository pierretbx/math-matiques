\section{Figures symétriques}

\begin{defn}
Deux figures sont symétriques par rapport à une droite si elles se superposent par pliage le long de cette droite. 

\medskip
\noindent Cette droite est appelée \dotfill
\end{defn}

\begin{ex}

\vspace{-3mm}
\noindent \begin{minipage}{8.5cm}
Les figures bleue et rouge se superposent par pliage le long de la droite $(d)$ donc elles sont symétriques par rapport à la droite $(d)$.

\medskip
\noindent On dit également que la figure rouge est le symétrique de la figure bleue dans la symétrie axiale d'axe $(d)$.

\medskip
\noindent Deux points sont symétriques par rapport à une droite s'ils se superposent par pliage le long de cette droite.

\medskip
\noindent Ici, les points $A$ et $M$ sont symétriques par rapport à la droite $(d)$.
\end{minipage}

\section{Symétrique d'un point}

\begin{defn}
Le symétrique d'un point A par rapport à une droite $(d)$ est le point $M$ tel que la droite $(d)$ soit la médiatrice du segment $[AM]$ (tel que $(d)$ soit la perpendiculaire au segment $[AM]$ en son milieu).
\end{defn}

\begin{rem}
Si un point appartient à une droite alors son symétrique par rapport à cette droite est le point lui-même.
\end{rem}

\begin{ex}

\section{Symétrique de figures usuelles et propriétés de la symétrie axiale}

\begin{prop}
Le symétrique d'une droite par rapport à un axe est une droite.

\medskip
\noindent La symétrie axiale conserve l'alignement.
\end{prop}

\begin{rem}
Le symétrique du milieu d'un segment est le milieu du segment symétrique.
\end{rem}

\begin{prop}
Le symétrique d'un cercle par rapport à un axe est un cercle de même rayon.

\medskip
\noindent Les centres des cercles sont symétriques par rapport à cet axe.
\end{prop}

\begin{prop}
La symétrie axiale conserve les mesures des angles, les périmètres et les aires.
\end{prop}
