\section{Notion de point}

\begin{defn}
Un point est un objet géométrique. C'est l'intersection de deux lignes. On le nomme par une lettre majuscule.
\end{defn}

\begin{ex}
Représenter un point $M$.

\noindent\grandscarreaux{\linewidth}{1.8cm}
\end{ex}

\begin{defn}
Deux points sont \dotfill s'ils occupent le même emplacement. 

\medskip \noindent Deux points sont \dotfill s'ils n'occupent pas le même emplacement.
\end{defn}

\begin{ex}
Représenter la situation suivante :
\begin{enumerate}
\item[•] les points $A$ et $B$ sont confondus ;
\item[•] les points $C$ et $D$ sont distincts.
\end{enumerate}

\medskip\noindent\grandscarreaux{\linewidth}{2cm}
\end{ex}

\section{Droite, demi-droite et segment}

\begin{center}
\renewcommand{\arraystretch}{1.28}
\begin{tabular}{|c|p{6.1cm}|c|}
\hline
Notation & \centering{Signification} & \hspace{2.1cm}Figure\hspace{2.1cm} \\\hline
$[AB]$ & Lire : « segment $AB$ » \hspace{3cm}\textcolor{white}{.} C'est le segment d'extrémités $A$ et $B$. &\\\hline
$(AB)$ & Lire « droite $AB$ » \hspace{3.5cm}\textcolor{white}{.} C'est la droite qui passe par les points $A$ et $B$. &\\\hline
$[AB)$ & Lire « demi-droite $AB$ ». \hspace{3.cm}\textcolor{white}{.} C'est la demi-droite d'origine $A$ passant par le point $B$. &\\\hline
$A \in(d)$ & Le point $A$ appartient à la droite $(d)$. & \\
& &\\[-4pt]\hline
$B \notin(D)$ & Le point $B$ n'appartient pas à la droite $(d)$. & \\
& &\\[-4pt]\hline
\end{tabular}
\end{center}

Une droite est un objet géométrique formé de points.

\noindent Une droite est illimitée.
\end{defn}
