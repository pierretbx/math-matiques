\chapter{Opérations avec les nombres décimaux}

\section{Ordre de grandeur}

\begin{defn}
Un ordre de grandeur d’un nombre est une valeur approchée simple de ce nombre.
\end{defn}

\begin{rem}
Calculer un ordre de grandeur permet de vérifier la cohérence d’un résultat.
\end{rem}

\begin{ex}
Détermine un ordre de grandeur de chaque calcul.
\begin{enumerate}
\item $546,3+52$
\item $65,7 \times 4,1$
\end{enumerate}

\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}

\begin{rem}
Un ordre de grandeur n'est pas unique.

\noindent Pour le deuxième exemple, on aurait pu prendre $70$ comme valeur proche de $65,7$ et $4$ comme valeur proche de $4,1$. Ce qui aurait donné $70 \times 4=280$ comme ordre de grandeur du produit $65,7\times 4,1$
\end{rem}

\section{Addition et soustraction de nombres décimaux}

\begin{prop}
Pour poser et effectuer une addition ou une soustraction de nombres décimaux, on place les nombres les uns en dessous des autres, de sorte que les virgules soient alignées verticalement.
\end{prop}

\begin{ex}
Calculer $15,2 + 0,57$ \quad et \quad $12-6,7$

\noindent\grandscarreaux{\linewidth}{4.8cm}
\end{ex}

\subsection{Multiplication de deux nombres décimaux}

\begin{prop}
Pour effectuer la multiplication de deux nombres décimaux,
\begin{enumerate}
\item[•]on effectue d'abord la multiplication sans tenir compte des virgules ;
\item[•]on place la virgule dans le produit en utilisant la méthode décrite ci-dessous.
\end{enumerate}
\end{prop}

\begin{ex}
Effectue la multiplication de $2,34$ par $1,2$

\noindent\grandscarreaux{\linewidth}{4.8cm}
\end{ex}

\section{Division d’un nombre décimal par un nombre entier}

\begin{prop}
Effectuer la division décimale de deux nombres, c'est trouver la valeur exacte ou une valeur approchée du quotient de ces deux nombres.
\end{prop}

\begin{ex}
Effectue la division de $75,8$ par $4$ puis celle de $4,9$ par $9$.

\noindent\grandscarreaux{\linewidth}{10cm}
\end{ex}
