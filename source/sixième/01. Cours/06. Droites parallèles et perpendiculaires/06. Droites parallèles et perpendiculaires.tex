\section{Définitions et notations}

\subsection{Droites sécantes}

\begin{defn}
Deux droites sécantes sont deux droites qui ont un seul point commun. Ce point est le \dotfill

\medskip \noindent \dotfill
\end{defn}

\begin{ex}

\subsection{Droites perpendiculaires}

\begin{defn}
Deux droites perpendiculaires sont deux droites qui se coupent en formant\dotfill

\medskip \noindent \dotfill
\end{defn}

\begin{ex}
Que peut-on dire des droites $(d)$ et $(d')$ ?

\begin{prop}
Le symbole $(d)\perp(d')$signifie que \og $(d)$ est perpendiculaire à $(d')$ \fg{}.
\end{prop}

\begin{rem}
\begin{enumerate}
\item[•] Deux droites perpendiculaires sont sécantes.
\item[•] Pour indiquer que deux droites sont perpendiculaires, on code un seul des quatre angles droites.
\item[•] On utilise une équerre pour tracer une droite perpendiculaire à une autre.
\end{enumerate} 
\end{rem}

Le symbole $//$ signifie \og est parallèle à \fg{}.
\end{prop}
