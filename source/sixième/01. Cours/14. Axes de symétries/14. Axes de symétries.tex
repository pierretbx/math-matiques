\chapter{Axes de symétries}

\section{Axe de symétrie d'une figure}

\begin{defn}
Une droite est un axe de symétrie d'une figure si les deux parties de la figure se superposent par pliage le long de cette droite.
\end{defn}

\begin{rem}
Une figure peut avoir aucun axe de symétrie, un seul axe de symétrie ou plusieurs axes de symétrie.
\end{rem}

\begin{ex}
Parmi les droites tracées en rouge, repasse en vert celles qui sont des axes de symétrie.

\begin{prop}
Pour vérifier qu'une droite est un axe de symétrie, on peut imaginer un pliage : chaque point de la figure doit se superposer avec un point de la même figure.
\end{prop}

\begin{ex}
Complète la figure pour que la droite $(d)$ soit un axe de symétrie.

\section{Axes de symétrie des figures usuelles}

\subsection{Figures sans axe ou avec un axe}
\enlargethispage{4cm}

\begin{prop}
\begin{enumerate}
\item[•] Un triangle quelconque n'a pas d'axe de symétrie en général.
\item[•] Un triangle isocèle possède un axe de symétrie.
\item[•] Un angle possède un axe de symétrie : sa bissectrice.
\end{enumerate}
\end{prop}

\begin{ex}
Trace l'axe de symétrie de chaque figure lorsqu'il existe.

\subsection{Quadrilatères et cercle}

\begin{prop}
\begin{enumerate}
\item[•] Un rectangle possède deux axes de symétrie.
\item[•] Un losange possède deux axes de symétrie.
\item[•] Un carré possède quatre axes de symétrie.
\item[•] Un cercle possède une infinité d'axes de symétrie : toutes les droites passant par son centre.
\end{enumerate}
\end{prop}

\section{Utiliser les axes de symétrie}

\begin{prop}
Si une figure possède un axe de symétrie, alors les points symétriques sont à la même distance de cet axe.
\end{prop}

\begin{ex}
La droite $(d)$ est un axe de symétrie de la figure. Place les points $B'$ et $C'$, symétriques des points $B$ et $C$ par rapport à $(d)$, puis termine la figure.

\smallskip
\begin{center}
\begin{tikzpicture}[scale=0.57,line cap=round,line join=round]
\draw[lightgray] (-5,-3) grid (5,3);
\draw[red,line width=1.3pt] (0,-3) -- (0,3);
\node[red,anchor=west] at (0,2.65) {$(d)$};
\draw[blue,line width=1.2pt] (-4,-2)--(-2,1)--(-1,0);
\fill[blue] (-4,-2) circle (2pt) node[below left,black] {$A$};
\fill[blue] (-2,1) circle (2pt) node[above left,black] {$B$};
\fill[blue] (-1,0) circle (2pt) node[below left,black] {$C$};
\draw[dashed,blue] (-4,-2)--(4,-2);
\fill[blue] (4,-2) circle (2pt) node[below right,black] {$A'$};
\end{tikzpicture}
\end{center}
\end{ex}
