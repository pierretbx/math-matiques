\section*{Devoir surveillé 17 - Les équations}
\addcontentsline{toc}{section}{Devoir surveillé 17 - Les équations}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Tester]
Le nombre $4$ est-il solution de l'équation $3x-5=7$ ? Justifier.
\medskip\noindent\grandscarreaux{\linewidth}{2.5cm}
\end{ex}
\begin{ex}[Résoudre]
Résoudre les équations : $5x+3=18$, $7-2x=15$ et $3(x+4)=24$.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
\begin{ex}[Problème]
Un nombre augmenté de son triple vaut $52$. Déterminer ce nombre en posant une équation.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
