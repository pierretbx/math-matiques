\paragraph{Exercice 2}: \hfill \textbf{3 points}

\noindent Exprime les grandeurs suivantes sous la forme d'une puissance de $10$.

\item $\np{100}$Mo = \hspace{1.5cm} o &

\item $\np{1}$dag = \hspace{1.5cm} g &

\end{tabularx}
\end{enumerate}

\begin{enumerate}
\item Faire une figure.
