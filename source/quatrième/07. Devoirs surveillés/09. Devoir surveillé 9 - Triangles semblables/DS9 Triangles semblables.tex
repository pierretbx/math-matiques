\section*{Devoir surveillé 9 - Triangles semblables}
\addcontentsline{toc}{section}{Devoir surveillé 9 - Triangles semblables}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Reconnaître]
Deux triangles ont pour angles $35^\circ$, $65^\circ$, $80^\circ$ et $65^\circ$, $80^\circ$, $35^\circ$. Sont-ils semblables ? Justifier.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Rapport]
Deux triangles semblables ont un rapport d'agrandissement $1,5$. Une longueur de $8$ cm devient quelle longueur ?
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Longueur manquante]
Dans deux triangles semblables, deux côtés correspondants mesurent $6$ cm et $9$ cm. Un autre côté du petit triangle mesure $10$ cm. Calculer le côté correspondant du grand triangle.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
