\section*{Devoir surveillé 14 - Multiplication et divisions de fractions}
\addcontentsline{toc}{section}{Devoir surveillé 14 - Multiplication et divisions de fractions}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Calculs]
Calculer et simplifier : $\dfrac{3}{5}\times\dfrac{10}{9}$, $\dfrac{7}{4}+\dfrac{5}{8}$ et $\dfrac{5}{6}\div\dfrac{10}{3}$.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
\begin{ex}[Comparer]
Comparer en justifiant : $\dfrac{3}{4}$ et $\dfrac{5}{8}$.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Problème]
Une recette utilise $\dfrac{3}{4}$ L de lait pour $6$ personnes. Quelle quantité faut-il pour $10$ personnes ?
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
