\section*{Devoir surveillé 19 - Les translations}
\addcontentsline{toc}{section}{Devoir surveillé 19 - Les translations}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Construction]
Construire l'image d'un triangle $ABC$ par la translation qui transforme $A$ en $A'$.
\medskip\noindent\grandscarreaux{\linewidth}{6cm}
\end{ex}
\begin{ex}[Propriétés]
Citer deux grandeurs conservées par une translation.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Raisonner]
Si la translation transforme $M$ en $M'$, expliquer pourquoi les segments $[AM]$ et $[A'M']$ sont parallèles et de même longueur.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
