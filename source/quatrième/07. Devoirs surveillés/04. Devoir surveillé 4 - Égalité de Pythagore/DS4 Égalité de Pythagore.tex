\section*{Devoir surveillé 4 - Égalité de Pythagore}
\addcontentsline{toc}{section}{Devoir surveillé 4 - Égalité de Pythagore}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Calculer]
Le triangle $ABC$ est rectangle en $A$, $AB=6$ cm et $AC=8$ cm. Calculer $BC$.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Tester]
Un triangle a pour côtés $7$ cm, $24$ cm et $25$ cm. Est-il rectangle ? Justifier.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Problème]
Une échelle de $5$ m est posée contre un mur. Son pied est à $1,4$ m du mur. Calculer la hauteur atteinte, arrondie au centimètre.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
