\section*{Devoir surveillé 16 - Divisibilité et nombres premiers}
\addcontentsline{toc}{section}{Devoir surveillé 16 - Divisibilité et nombres premiers}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Divisibilité]
Dire si $372$ est divisible par $2$, $3$, $4$, $5$ et $9$. Justifier chaque réponse.
\medskip\noindent\grandscarreaux{\linewidth}{3.5cm}
\end{ex}
\begin{ex}[Nombres premiers]
Décomposer $756$ en produit de facteurs premiers, puis rendre irréductible $\dfrac{252}{756}$.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
\begin{ex}[Problème]
Deux bus partent du même arrêt. Le premier passe toutes les $12$ minutes et le second toutes les $18$ minutes. Ils partent ensemble à $8$ h. À quelle heure repasseront-ils ensemble pour la première fois ?
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
