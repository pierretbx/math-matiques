\section*{Pyramide et cône 2 \hspace{2cm} Sujet A}
\addcontentsline{toc}{section}{Pyramide et cône 2 Sujet A}

\noindent \textit{L'usage de la calculatrice est \textbf{autorisée}.} \hfill \textit{Durée : 55 minutes}

\noindent \textit{La rédaction sera évaluée au même titre que les résultats, on prendra donc soin de justifier chacun de nos calculs.} 

\paragraph{Exercice 1}: \hfill \textbf{4 points}

\noindent On considère le cube ABCDEFGH de coté $4$cm. 

\noindent On note $I$ le milieu du segment $[AD]$.

\vspace{-15mm}
\noindent \begin{minipage}{10cm}

\vspace{8mm}
\begin{enumerate}
\item Justifie que la hauteur de la pyramide $IDHGC$ mesure $2$cm
\item Calcule le volume de la pyramide $IDHGC$, arrondi au cm$^3$.
\item Construis le patron de cette pyramide.
\end{enumerate}
\end{minipage}
\begin{minipage}{6cm}
\begin{center}
\begin{tikzpicture}[general,scale=0.6]
\draw(1,1.5) node[below] (A){$A$};
\draw(5,1) node[below] (B){$B$};
\draw(7,2) node[right] (C){$C$};
\draw(3,2.5) node[above left] (D){$D$};
\draw (1,5.5) node[left] (E){$E$};
\draw(5,5) node[above] (F){$F$};
\draw(7,6) node[right] (G){$G$};
\draw(3,6.5) node[above] (H){$H$};
\draw(2,2) node[below] (I){$I$};
\fill [color=exer2!20](I.north)--(H.south)--(G.west);
\fill [color=exer2!20](I.north)--(G.west)--(C.west);
\draw(3,2.5) node[above left] {$D$};
\draw [color=red, dashed] (I.north) -- (H.south);
\draw [color=red, dashed] (I.north) -- (C.west);
\draw [color=red, dashed] (I.north) -- (G.west);
\draw (1,1.5)-- (5,1)-- (5,5)--(1,5.5)--cycle;
\draw (1,5.5)--(3,6.5)--(7,6)--(5,5);
\draw (7,6)--(7,2)--(5,1)--(5,5);
\draw [dashed] (7,2)--(3,2.5);
\draw [dashed] (3,2.5)-- (1,1.5);
\draw [color=red, dashed] (3,2.5)-- (2,2);
\draw [dashed] (3,2.5)-- (3,6.5);
\draw (1,5.5)--(5,5)--(7,6);
\end{tikzpicture}
\end{center}
\end{minipage}

\includegraphics[scale=1]{../40.Image/DS8_Pyramide_et_cone_2_3.png}

\hspace{5mm}
\noindent \begin{minipage}{12cm}
\noindent On sait que $BJ = 18$cm, $FJ = 14,4$cm, $BH = 12,5$cm et $FC = 10$cm
\begin{enumerate}
\item  Calculer $V_1$ le volume exact du grand cône (dont la base a pour rayon $BH$ ).
\item Calculer $V_2$ le volume exact du petit cône (dont la base a pour rayon $FC$ ).
\item En déduire $V_3$ le volume du tronc de cône (la partie colorée). Donner la valeur exacte puis un arrondi au cm$^3$ près.
\item En utilisant le théorème de Pythagore dans deux triangle différent, calculer la longueur $CH$ .
\end{enumerate}
\end{minipage}

\paragraph{Exercice 5}: \hfill \textbf{4 points}

\noindent Un réservoir d'eau est constitué d'une partie cylindrique et d'une partie conique.

\vspace{-10mm}
\noindent \begin{minipage}{11.5cm}
\vspace{8mm}
\begin{enumerate}
\item  Donne la valeur exacte du volume
de ce réservoir.
\item Ce réservoir peut-il contenir $1500$L ?

\medskip
\textit{Question bonus 1 point}
\item Si ce réservoir contient $1000L$, à quelle hauteur par rapport au sommet du cône arrivera l'eau ?
\end{enumerate}
