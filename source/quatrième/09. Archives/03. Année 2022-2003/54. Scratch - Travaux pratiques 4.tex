\noindent \textbf{Objectif 2  :} \quad Le bon nombre

\medskip
\noindent Le jeu du bon nombre consiste à faire deviner un nombre précis à un joueur.

\noindent Le jeu doit mettre en interaction le lutin et le joueur :
\begin{enumerate}
\item[•] le lutin pense à un nombre choisi au hasard entre $1$ et $10$ ;
\item[•] l'utilisateur doit deviner ce nombre. Pour cela, il n'a droit qu'à un seul essai.
\end{enumerate}

\medskip
\begin{dinglist}{111}
\item Faire dire au lutin \og Tu vas essayer de deviner un nombre entier compris entre $1$ et $10$ \fg{} pendant $2$ secondes.
\item Créer une variable \ovalvariable{nombre} et la cacher pour qu'elle n'apparaisse pas à l'écran.
\item Affecter à la variable \ovalvariable{nombre} une valeur aléatoire entre $1$ et $10$ : \ovaloperator{nombre aléatoire entre \ovalnum{1} et \ovalnum{10}} 
\item Demander \og Quel nombre proposes-tu ? \fg{} 
\item \textbf{Si} la réponse donnée est correct, \textbf{alors} faire dire au lutin \og Bravo ! Tu as deviné du premier coup! \fg{}. 

\textbf{Sinon}, faire dire au lutin \og Perdu le nombre était : \fg{} et faire afficher le bon nombre   
\end{dinglist}

\ding{226} \textit{Appelle le professeur pour la vérification}
