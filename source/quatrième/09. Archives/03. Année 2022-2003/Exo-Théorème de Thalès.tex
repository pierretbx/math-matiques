\begin{exer1}

\begin{enumerate}[font=\color{exer1}]
\item Fais un dessin en vraie grandeur et code-le.
\item Montre que $(LS$) est parallèle à $(OR)$.
\item Déduis-en que $(LS)$ est perpendiculaire à $(AO)$.
\end{enumerate}
\end{exer1}
