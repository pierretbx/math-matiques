\begin{dft}
Soit deux points $P$ et $P'$.

\noindent On appelle translation qui transforme $P$ en $P'$, le glissement :
\begin{enumerate}
\item[•] selon la direction de la droite $(PP’)$,
\item[•] dans le sens de $P$ vers $P'$,
\item[•] d'une longueur égale à $PP'$.
\end{enumerate}

\noindent La figure F' est l'image de la figure $F$ par cette translation.

\vspace{-3.9cm}
\begin{flushright}
\includegraphics[scale=0.3]{../40.Image/19. Translation_2.png}
\end{flushright}
\end{dft}

\begin{rmq}
Pour schématiser la translation, on peut tracer une flèche allant de $P$ vers $P’$.
\end{rmq}

\section{Constructions définies par une translation}

\vspace{-3mm}
\subsection{Sur papier quadrillé}

\vspace{-1mm}
\begin{meth}[Construire l'image d'une figure par une translation sur papier quadrillé]

\noindent Soit la translation qui transforme $A$ en $A'$ schématisée par la flèche rouge.

\noindent Construire l'image du quadrilatère $BCDE$ par cette translation.

\noindent Pour construire l'image du point $B$, on \og reproduit \fg{} la flèche \\rouge en plaçant son origine en $B$.

\noindent Pour reproduire la flèche rouge, on peut s'aider du quadrillage.

\includegraphics[scale=0.4]{../40.Image/19. Translation_4.png}

\vspace{-5cm}
\begin{flushright}
\includegraphics[scale=0.3]{../40.Image/19. Translation_3.png}
\end{flushright}

\noindent On obtient le point $B'$ tel que les deux flèches rouges aient la même direction, le même sens et la même longueur.

\noindent On refait de même pour les autres points et on obtient l'image $B’C’D’E’$ du quadrilatère $BCDE$ par la translation.
\end{meth}
