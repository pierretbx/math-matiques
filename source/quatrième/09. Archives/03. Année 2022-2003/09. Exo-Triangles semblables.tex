\begin{exer2}
On considère deux droites parallèles $(d_1)$ et $(d_2)$. 

\noindent $M$ est un point de $(d_1)$ et $N$ est un point de $(d_2)$. 

\noindent Une droite $(d_3)$ passe par le milieu $O$ de $[MN]$ et coupe $(d_1)$ en $R$ et $(d_2)$ en $S$.

\vspace{-3mm}
\begin{center}
\includegraphics[scale=0.3]{../40.Image/09.Exo-Triangles_semblables_6.png}

\vspace{-3mm}
\begin{enumerate}[font=\color{exer2}]
\item Prouve que les triangles $ROM$ et $NOS$ sont égaux.
\item Déduis-en que $O$ est le milieu de $[RS]$.
\end{enumerate}
\end{exer2}

\begin{exer3}
$ABC$ est un triangle isocèle en $C$. 

\noindent La médiatrice de $[BC]$ coupe la droite $(AB)$ en $M$.
 
\noindent Sur la droite $(MC)$, on a placé le point $N$ de telle sorte que $CN=AM$.

\vspace{-6mm}
\noindent \begin{minipage}{11cm}
\begin{enumerate}[font=\color{exer3}]
\item Démontre que les angles $\widehat{MBC}$ et $\widehat{MCB}$ ont la même mesure.
\item Démontre que les angles $\widehat{MAC}$ et $\widehat{NCB}$ ont la même mesure.
\item Déduis-en que les triangles $AMC$ et $CNB$ sont égaux.
\item Démontre que le triangle $MBN$ est isocèle.
\end{enumerate}
\end{minipage}
