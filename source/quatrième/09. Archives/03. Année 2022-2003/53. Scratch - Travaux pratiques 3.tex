\noindent \textbf{Objectif 1 :} \quad Frise Carrés

\medskip
\noindent \begin{minipage}{13.5cm}
\begin{dinglist}{111}
\item Crée un bloc nommé Carré qui permet de tracer un carré de côté $100$.
\item En utilisant les blocs, écris un programme permettant de tracer la frise ci-dessous.
\end{dinglist}
\begin{center}
\includegraphics[scale=0.5]{../40.Image/53.Frises_1.png}
\end{center}

\vspace{-5mm}
\ding{226} \textit{Appelle le professeur pour la vérification}
\end{minipage}
\begin{minipage}{5cm}
\begin{scratch}[scale=0.7]
\initmoreblocks{définir \namemoreblocks{Carré}}
\blockrepeat{répéter \ovalnum{4} fois}
{
\blockmove{avancer de \ovalnum{100}}
\blockmove{tourner \turnleft{} de \ovalnum{90} degrés}
}
\end{scratch}
\end{minipage}
