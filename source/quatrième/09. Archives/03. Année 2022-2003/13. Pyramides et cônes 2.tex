\section{Volume d'une pyramide ou d'un cône}

\begin{prop}
Pour calculer le volume d'une pyramide ou d'un cône de révolution, on utilise la même formule :
$$V=\dfrac{\text{Aire de la base} \times \text{Hauteur}}{3}$$
\end{prop}

\begin{ex}
Calcule le volume d'une pyramide de hauteur $146$m ayant pour base un carré de coté $230$m

\end{ex}

\begin{ex}

\end{ex}
