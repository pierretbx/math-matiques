\chapter{Multiplication et division de fraction}

\section{Rappel : Égalité de quotients}

\subsection{Simplification de quotient}

\begin{memento}
Si on multiplie ou si on divise le numérateur et le dénominateur d'un quotient par un même nombre non nul alors on obtient un quotient égal.

\medskip
\noindent Pour tous nombres $a$, $b$ et $k$ où $b$ et $k$ sont non nuls :
$$\dfrac{a \times k}{b \times k} = \dfrac{a}{b} \hspace{1cm} \text{ et } \hspace{1cm} \dfrac{a \div k}{b \div k} = \dfrac{a}{b}$$
\end{memento}

\begin{ex}
Simplifier le quotient $\dfrac{42}{-140}$

\noindent\grandscarreaux{\linewidth}{3.2cm}
\end{ex}

\begin{ex}
Détermine le nombre manquant dans l'égalité $\dfrac{-1,2}{6}=\dfrac{\dots}{18}$

\noindent\grandscarreaux{\linewidth}{3.2cm}
\end{ex}

\subsection{Produit en croix}

\begin{prop}
Si deux nombres en écriture fractionnaire sont égaux alors leurs produits en croix sont égaux.

\medskip
\noindent Réciproquement : si les produits en croix de deux nombres en écriture fractionnaire sont égaux alors ces deux nombres sont égaux.
\end{prop}

\begin{rem}
Pour démontrer que deux nombres en écriture fractionnaire ne sont pas égaux, il suffit de démontrer que leurs produits en croix ne sont pas égaux.
\end{rem}

\section{Rappel : Addition ou soustraction}

\begin{memento}
Pour additionner (ou soustraire) des nombres en écriture fractionnaire ayant le même dénominateur, on additionne (ou on soustrait) les numérateurs et on garde le dénominateur commun.

\medskip
\noindent Pour tous nombres $a$, $b$ et $c$ où $b$ est non nul :
$$\dfrac{a}{b}+\dfrac{c}{b} = \dfrac{a+c}{b}$$
\end{memento}

\begin{rem}
Si les nombres en écriture fractionnaire n'ont pas le même dénominateur, il faut les réduire au même dénominateur.
\end{rem}

\begin{ex}
Calcule l'expression suivante $A=-1+\dfrac{13}{20}-\dfrac{1}{10}$.

\noindent\grandscarreaux{\linewidth}{5.6cm}
\end{ex}

\section{Multiplication}

\begin{memento}
Pour multiplier des nombres en écriture fractionnaire, on multiplie les numérateurs entre eux et les dénominateurs entre eux.

\medskip
\noindent Pour tous nombres $a$, $b$, $c$ et $d$ où $b$, $c$ et $d$ non nuls :
$$\dfrac{a}{b} \times \dfrac{c}{d} = \dfrac{a \times c}{b \times d}$$
\end{memento}

\begin{ex}
Calcule l'expression $B=-\dfrac{35}{33} \times \dfrac{-39}{-80}$

\noindent\grandscarreaux{\linewidth}{4.8cm}
\end{ex}

\section{Division de deux quotients}

\subsection{Inverse d'un nombre non nul}

\begin{defn}
Deux nombres sont inverses l'un de l'autre si leur produit est égal à $1$.
\end{defn}

\begin{prop}
Tout nombre $x$ non nul admet un inverse (noté $x^{-1}$) qui est le nombre $\dfrac{1}{x}$

\noindent Tout nombre en écriture fractionnaire $\dfrac{a}{b}$($a\neq 0$ et $b \neq 0$) admet un nombre inverse qui est le nombre $\dfrac{b}{a}$.
\end{prop}

\begin{rem}
\begin{enumerate}
\item[•] Un nombre et son inverse ont toujours le même signe.
\item[•] Zéro est le seul nombre qui n'admet pas d'inverse.
\end{enumerate}
\end{rem}

\begin{ex}
Quels sont les inverses des nombres $3$ et $\dfrac{-3}{7}$ ?

\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex}

\subsection{Diviser des quotients}

\begin{memento}
Diviser par un nombre non nul revient à multiplier par l'inverse de ce nombre.

\medskip
\noindent 
Pour tous nombres $a$, $b$, $c$ et $d$ où $b$, $c$ et $d$ non nuls :
$$\dfrac{a}{b} \div \dfrac{c}{d} = \dfrac{a}{b} \times \dfrac{d}{c}  \hspace{1cm} \text{ ou } \hspace{1cm} \dfrac{\quad \dfrac{a}{b} \quad }{\dfrac{c}{d}}= \dfrac{a}{b} \times \dfrac{d}{c} $$
\end{memento}

\begin{ex}
Calcule $C=\dfrac{-8}{7} \div \dfrac{5}{-3}$

\noindent\grandscarreaux{\linewidth}{6.4cm}
\end{ex}

\begin{ex}
Calcule $D=\dfrac{\dfrac{-32}{21}}{\dfrac{-48}{-35}}$ et donne le résultat en le simplifiant le plus possible.

\noindent\grandscarreaux{\linewidth}{6.4cm}
\end{ex}
