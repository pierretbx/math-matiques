\chapter{Pyramides et cônes}

\section{Pyramides et cônes : définition et perspective}

\subsection{La pyramide}

\begin{defn}
Une pyramide est un solide dont :
\begin{enumerate}
\item[•] une face est un polygone appelée la base de la pyramide ;
\item[•] les autres faces, appelées face latérales, sont des triangles qui ont un sommet commun, appelé le sommet de la pyramide.
\end{enumerate}
\end{defn}

\begin{defn}
La hauteur d'une pyramide est le segment issu de son sommet et perpendiculaire à la base.
\end{defn}

\begin{defn}
Une arête latérale est un segment joignant les sommets de la base au sommet de la pyramide.
\end{defn}

\begin{ex}

\noindent \begin{minipage}{5.5cm}
\includegraphics[scale=0.7]{Pyramide_et_cone_1.png}
\end{minipage}
\begin{minipage}{11cm}
Traçons une pyramide $SABCDE$ de sommet $S$ en perspective cavalière et décrivons les éléments de ce solide.
\medskip

\begin{enumerate}
\item[•] Le sommet de cette pyramide est le point $S$.
\item[•] La base de cette pyramide est la pentagone $ABCDE$.
\item[•] Les faces latérales sont les triangles : $SAB$, $SBC$, $SCD$, $SDE$, $SEA$.
\item[•] Les arêtes latérales sont les segments : $[AS]$, $[BS]$, $[CS]$, $[DS]$, $[ES]$.
\item[•] La hauteur de la pyramide est le segment $[OS]$.
\end{enumerate}
\end{minipage}
\end{ex}

\begin{defn}
Une pyramide régulière est une pyramide dont la base est un polygone régulier (par exemple un triangle équilatéral ou un carré) et dont les faces latérales sont des triangles isocèles superposables.
\end{defn}

\noindent \begin{minipage}{11cm}
\begin{rem}\medskip
\begin{enumerate}
\item[•] Une pyramide régulière à base triangulaire s'appelle un tétraèdre.
C'est un solide dont les quatre faces sont des triangles équilatéraux superposables.\medskip
\item[•] La hauteur d'une pyramide régulière passe par le centre de la base qui est le point de concours des diagonales.
\end{enumerate} 
\end{rem}
\end{minipage}

\section{Le cône de révolution}

\vspace{-8mm}
\noindent \begin{minipage}{12cm}
\begin{defn}
Un cône de révolution est un solide qui est généré par un triangle rectangle en rotation autour d'un des côtés de son angle droit.
\end{defn}

\begin{defn}
La base du cône de révolution est un disque.
\end{defn}

\begin{defn}
La hauteur du cône de révolution est le segment qui joint le centre de ce disque au sommet du cône. Il est perpendiculaire au disque de base.
\end{defn}
\end{minipage}

\begin{rem}
La surface latérale d'un cône, appelée aussi développement, est générée par l'hypoténuse du triangle rectangle. Elle a la forme d'un secteur de disque.
\end{rem}
