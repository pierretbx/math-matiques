\section{Développement}

\vspace{-2mm}
\subsection{La distributivité simple}

\vspace{-2mm}
\begin{prop}
Développer une expression littéral, c'est transformer un produit en somme. 

\noindent Pour le faire, on utilise la règle de la distributivité : 

\medskip
\noindent Pour tous nombres relatifs $k$, $a$ et $b$ : \hspace{2cm} $k \times (a+b) = k \times a+k \times b $
\end{prop}

\vspace{-2mm}
\begin{ex}
Développe les expressions suivantes : 

\hspace{0.5cm}\begin{minipage}{6cm}
$S=2(x+7)$
\end{minipage}
\begin{minipage}{6cm}
$O=3x(1+2x)$
\end{minipage}
\begin{minipage}{6cm}
$L=(x^2-3)\times 5$
\end{minipage}

\noindent\grandscarreaux{\linewidth}{1.6cm}
\end{ex}

\vspace{-2mm}
\subsection{Suppression de parenthèses dans une somme algébrique}

\vspace{-2mm}
\begin{defn}
Une somme algébrique est un calcul dans lequel il n'y a que des additions et des soustractions.
\end{defn}

\vspace{-2mm}
\begin{ex}
Entoure la ou les sommes algébriques

\medskip
\noindent \begin{minipage}{6cm}
$L=3x+1-2x^2$
\end{minipage}
\begin{minipage}{6cm}
$I=x-(4+2x)-7$
\end{minipage}
\begin{minipage}{6cm}
$T=3(4x-5)+10$
\end{minipage}
\end{ex}

\vspace{-2mm}
\begin{prop}[Addition et parenthèses]

Quels que soient les nombres $a$ et $b$, on a : \hspace{1cm}$ \dots + ( a + b ) \quad = \quad \dots + \textcolor{red}{1 \times} ( a + b ) \quad = \quad \dots + a + b $
\end{prop}

\vspace{-2mm}
\begin{ex}
Développe les expressions suivantes : 

\hspace{0.5cm}\begin{minipage}{5cm}
$M = 5 + (7 + x)$
\end{minipage}
\begin{minipage}{7cm}
$I = (4 - 3x) + (-2x + 6)$
\end{minipage}
\begin{minipage}{6cm}
$E = 4x + (3x - 5)$
\end{minipage}

\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex} 
  
\begin{prop}[Soustraction et parenthèses]

Quels que soient les nombres $a$ et $b$, on a : \hspace{1cm}$ \dots - ( a + b ) = \dots + \textcolor{red}{(-1) \times} ( a + b ) = \dots + (-a) + (-b) $
\end{prop}

\vspace{-2mm}
\begin{ex}
Développe les expressions suivantes : 

\hspace{0.5cm}\begin{minipage}{5cm}
$R = 5 - (7 + x)$
\end{minipage}
\begin{minipage}{7cm}
$A = (4 - 3x) - (-2x + 6)$
\end{minipage}
\begin{minipage}{6cm}
$T = 4x - (3x - 5)$
\end{minipage}

\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex} 
  
\subsection{Cas plus complexes} 

\noindent Pour développer et réduire des expressions littérales plus complexes, il faut  prendre en compte les règles de priorité.

\begin{ex}
Développe les expressions suivantes : 

\hspace{0.5cm}\begin{minipage}{8cm}
$R = (x-5) + 3(x + 4)$
\end{minipage}
\begin{minipage}{8cm}
$E = 3 - (2x + 1) \times  5$
\end{minipage}

\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex}
