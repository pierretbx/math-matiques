\chapter{Produit de nombres relatifs}

\section{Addition de nombres relatifs}

\begin{memento}
\begin{enumerate}
\item[•] Pour additionner deux nombres relatifs de même signe, on additionne leur distance à zéro et on garde le signe commun.
\item[•] Pour additionner deux nombres relatifs de signes contraires, on soustrait la plus petite distance à zéro de la plus grande et on prend le signe de celui qui a la plus grande distance à zéro.
\end{enumerate}
\end{memento}

\begin{meth}[Addition de relatifs de même signe]
On veut effectuer l'addition suivante : $A=(-2)+(-3)$.

\renewcommand{\arraystretch}{1.3}
\begin{tabularx}{\linewidth}{p{4cm}*{1}{>{ \arraybackslash}X}}
$A = (-2)+(-3)$ & On veut additionner deux nombres relatifs de même signe.\\
$A = -(2+3)$ &  On additionne leur distance à zéro et on garde le signe commun : \og - \fg{}\\
$A= -5$ & On calcule.\\
\end{tabularx}
\end{meth}

\begin{meth}[Addition de relatifs de signe contraire]
On veut effectuer l'addition suivante : $B=(-5)+(+7)$.

\renewcommand{\arraystretch}{1.3}
\begin{tabularx}{\linewidth}{p{4cm}*{1}{>{ \arraybackslash}X}}
$B=(-5)+ (+7)$ & On veut additionner deux nombres relatifs de signes        			contraires.\\
$B= + (7-5)$ & On soustrait leur distance à zéro et on écrit le signe du nombre qui a la plus grande distance à zéro $(+7)$.\\
$B= +2$ & On calcule.\\
\end{tabularx}
\end{meth}

\section{Soustraction de deux nombres relatifs}

\begin{defn}
L'opposé d'un nombre relatif est le nombre de signe contraire qui a la même distance à zéro.
\end{defn}

\begin{ex}
Donne l'opposé de chacun des nombres relatifs : \quad $-2 531$ ; \quad $0$ ; \quad $1 245$ ; \quad $-0,03$ \quad et \quad $+ 0,003$.

\noindent\grandscarreaux{\linewidth}{1.6cm}
\end{ex}

\begin{memento}
Soustraire un nombre relatif revient à ajouter son opposé.
\end{memento}

\begin{meth}[Soustraction de relatifs]
On veut effectuer la soustraction suivante : $C=(-2)-(-3)$.

\renewcommand{\arraystretch}{1.3}
\begin{tabularx}{\linewidth}{p{4cm}*{1}{>{ \arraybackslash}X}}
$C=(-2)-(-3)$ & On veut soustraire le nombre $-3$.\\
$C=(-2)+(+3)$ & On ajoute l'opposé de $-3$ qui est $+3$.\\
$C=+(3-2)$ & On ajoute deux nombres de signes contraires donc on soustrait leur distance à zéro et on prend le signe du nombre qui a la plus grande distance à zéro $(+3)$.\\
$C= +1$ & On calcule.\\
\end{tabularx}
\end{meth}

\section{Multiplication de nombres relatifs}

\vspace{-2mm}
\subsection{Produit de deux facteurs}

\begin{memento}
Pour multiplier deux nombres relatifs, on multiplie leur distance à zéro et on applique la règle des signes :
\begin{enumerate}
\item[•] le produit de deux nombres relatifs, de même signe est positif ;
\item[•] le produit de deux nombres relatifs de signes contraires est négatif.
\end{enumerate}
\end{memento}

\begin{ex}
Effectue la multiplication $F=(-4)\times  (-2,5)$.

\noindent\grandscarreaux{\linewidth}{2.2cm}
\end{ex}

\begin{ex}
Effectue la multiplication : $G=0,2 \times (-14)$

\noindent\grandscarreaux{\linewidth}{2.2cm}
\end{ex}

\begin{prop}
Multiplier un nombre relatif par $-1$ revient à prendre son opposé.
\end{prop}

\begin{rem}
Cela signifie que pour tout nombre relatif $a$ : $-1 \times a=-a$
\end{rem}

\vspace{-2mm}
\subsection{Produit de plusieurs facteurs}

\begin{meth}
Pour déterminer le signe d'un produit de plusieurs facteurs, on compte le nombre de facteurs négatifs.
Lorsque le nombre de facteurs négatifs de ce produit est pair, le produit est positif.
Lorsque le nombre de facteurs négatifs de ce produit est impair, le produit est négatif.
\end{meth}

\begin{ex}
Quel est le signe du produit : $H=-6 \times 7 \times (-8) \times (-9)$ ?

\noindent\grandscarreaux{\linewidth}{1.6cm}
\end{ex}

\begin{ex}
Calcule le produit : $J=2 \times (-4) \times (-5) \times (-2,5) \times (-0,8)$

\noindent\grandscarreaux{\linewidth}{2.2cm}
\end{ex}
