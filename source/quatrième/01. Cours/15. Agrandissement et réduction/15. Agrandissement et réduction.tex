\chapter{Agrandissement et réduction}

\section{Coefficient d'agrandissement ou de réduction}

\begin{defn}
Lorsque deux figures ont la même forme et des longueurs proportionnelles on dit que l'une est un agrandissement ou une réduction de l'autre. 
\end{defn}

\begin{prop}
Le coefficient de proportionnalité $k$ permettant de passer d'une figure à l'autre est défini par : 

\medskip
\noindent \begin{minipage}{9cm}
$$k \quad = \quad \dfrac{\text{longueur d'un côté de la figure d'arrivée}}{\text{longueur d'un côté de la figure de départ}}$$
\end{minipage}
\hspace{5mm}
\begin{minipage}{6cm}
\begin{enumerate}
\item[•] Si $0 < k < 1$, c'est une réduction
\item[•] Si $k>1$, c'est un agrandissement
\end{enumerate}
\end{minipage}


\end{prop}

\begin{rem}
Dans un agrandissement ou une réduction, les angles sont conservés. 
\end{rem}

\begin{ex}
On sait que le quadrilatère $MNOP$ est une réduction du quadrilatère $ABCD$ de rapport $0,6$ (cela signifie que le coefficient de réduction est $0,6$).

\section{Effet sur l'aire et le volume}

\begin{prop}
Pour un agrandissement ou une réduction de rapport $k$ : 
\begin{enumerate}
\item[•] les aires sont multipliées par $k^2$
\item[•] les volumes sont multipliés par $k^3$
\end{enumerate}
\end{prop}
