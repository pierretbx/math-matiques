\chapter{Proportionnalité}

\section{Grandeurs proportionnelles}

\subsection{Reconnaître un tableau de proportionnalité}

\begin{defn}
Un tableau de nombres relève d'une situation de proportionnalité si un même coefficient (non nul) multiplicateur s'applique dans tout le tableau. On parle alors de coefficient de proportionnalité.
\end{defn}

\begin{ex}
Ces tableaux de nombres sont-ils des tableaux de proportionnalité ?

\begin{enumerate}
\item \begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|*{5}{>{\centering \arraybackslash}X|}}
\hline
$12$ & $18$ & $32$ & $27$ & $54$\\\hline
$8$ & $12$ & $20$ & $18$ & $36$\\\hline
\end{tabularx}
\end{center}
\noindent\grandscarreaux{\linewidth}{3.2cm}

\item \begin{center}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{|*{5}{>{\centering \arraybackslash}X|}}
\hline
$5$ & $8$ & $14$ & $19$ & $24$\\\hline
$12$ & $19,2$ & $33,6$ & $45,6$ & $57,6$\\\hline
\end{tabularx}
\end{center}
\noindent\grandscarreaux{\linewidth}{3.2cm}
\end{enumerate}
\end{ex}

\subsection{Quatrième proportionnelle}

\begin{prop}
Dans une situation de proportionnalité, la quatrième proportionnelle est le nombre \og $x$ \fg{}  calculé à partir de $3$ autres nombres déjà connus ($a$, $b$ et $c$). 

\medskip 
\noindent Le tableau ci-dessous est un tableau de proportionnalité.

\vspace{-10mm}
\begin{flushright}
\renewcommand{\arraystretch}{2}
\begin{tabularx}{6cm}{|*{2}{>{\centering \arraybackslash}X|}}
\hline
$a$ & $c$ \\\hline
$b$ & $x$ \\\hline
\end{tabularx}
\end{flushright}

\vspace{-11mm}
\noindent On a : \quad $\dfrac{a}{b}=\dfrac{x}{c}$ \quad avec $a$, $b$ et $c$ sont différents de zéro.

\medskip 
\noindent Et donc : \quad $a\times x=b\times c$ \quad (égalités des produits en croix)
\end{prop}

\begin{ex}
Calcule le prix $x$ de trois baguettes grâce au tableau \\de proportionnalité suivant.

\vspace{-15mm}
\begin{flushright}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{7cm}{p{4cm}|*{2}{>{\centering \arraybackslash}X|}}
\hline
\cellcolor{lightgray!50}{Nombre de baguettes} & $5$ & $3$ \\\hline
\cellcolor{lightgray!50}{Prix en \euro} & $4,25$ & $x$ \\\hline
\end{tabularx}
\end{flushright}

\noindent\grandscarreaux{\linewidth}{2.4cm}
\end{ex}

\section{Caractérisation}

\subsection{Représentation Graphique}

\begin{prop}
Si on représente, dans un repère, une situation de proportionnalité alors on obtient des points alignés avec l'origine du repère.
\end{prop}

\begin{ex}
Le périmètre $p$ d'un carré est proportionnel à son côté $c$ puisqu'on a $p=4 \times c$.
Représente graphiquement le périmètre en fonction du côté.

\noindent\grandscarreaux{\linewidth}{3.2cm}
\end{ex}

\begin{prop}
Si une situation est représentée par des points alignés avec l'origine du repère alors c'est une situation de proportionnalité.
\end{prop}
