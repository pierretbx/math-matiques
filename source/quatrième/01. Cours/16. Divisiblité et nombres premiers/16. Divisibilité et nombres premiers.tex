\begin{meth}[Critère de divisibilité]
\begin{enumerate}
\item[•] Un nombre est divisible par $2$ si il se termine par $0$, $2$, $4$, $6$ ou $8$
\item[•] Un nombre est divisible par $5$ s'il se termine par $0$ ou $5$.

\item[•] Un nombre est divisible par $3$ si la somme de ses chiffres est divisible par $3$.

\medskip
\textit{Exemple :} $249$ est divisible par $3$ car $2 + 4 + 9 = 15$ qui est divisible par $3$ ($3 \times 5 = 15$).

\item[•] Un nombre est divisible par $9$ si la somme de ses chiffres est divisible par $9$.

\medskip
\textit{Exemple :} $2 538$ est divisible par $9$ car $2 + 5 + 3 + 8 = 18$ qui est divisible par $9$ ($3 \times 9 = 18$).
\end{enumerate}
\end{meth}

\section{Nombres premiers}

\vspace{-2mm}
\begin{defn}
Un nombre qui n'est divisible que par $1$ est lui-même est appelé un nombre premier.
\end{defn}

\begin{rem}
Il n'existe pas à ce jour de formule pour trouver des nombres premiers. Afin d'être certains qu'un nombre soit premier, il faut faire de nombreux calculs, souvent a l'aide d'ordinateur.
\end{rem}

\section{Décomposition en produit de facteurs premiers}

\vspace{-5mm}
\noindent \begin{minipage}{14cm}

\begin{prop}
On peut toujours décomposer un nombre non premier en produit de plusieurs facteurs premiers, cette décomposition est unique.
\end{prop}

\begin{ex}
La décomposition du produit en facteurs premier de $324$ est : $$324 = 2^2 \times 3^4$$
\end{ex}
