\section{Tester une égalité}

\begin{defn}
Une égalité est constituée de deux membres séparés par le signe « = » .
\end{defn}

\begin{ex}

\vspace{-5mm}
 $$\underset{\text{membre de gauche} }{\underbrace{5 \times 4}} \quad = \quad  \underset{\text{membre de droite} }{\underbrace{12 + 8}}$$
\end{ex}
 
\begin{prop}
Une égalité est vraie lorsque ses deux membres ont la même valeur.
\end{prop}

\begin{rem}Une égalité comportant des expressions littérales est vraie ou fausse selon les valeurs données à la lettre.
\end{rem}

\begin{ex}
\begin{enumerate}
\item[•] L'égalité $5 + x = 8$ est vraie pour $x = 3$. En effet, $5 + 3 = 8$
\item[•] L'égalité $5 + x = 8$ est fausse pour $x = 4$. En effet, $5 + 4 = 9$ et $9 \times 8$.
\end{enumerate}
\end{ex}

\begin{meth}[Tester une égalité]

Pour tester si une égalité est vraie :
\begin{enumerate}
\item On calcule la valeur du membre de gauche en remplaçant chaque lettre par le nombre donné.
\item On calcule la valeur du membre de droite en remplaçant chaque lettre par le nombre donné.
\item On observe l'égalité ou non des deux valeurs obtenues et on conclut.
\end{enumerate}
\end{meth}

\begin{ex}
On considère l'expression $3 \times x + 5 = 5 \times x – 9$.

\medskip
\noindent \begin{minipage}{8cm}
Cette égalité est-elle vraie pour $x = 2$ ?
\begin{enumerate}
\item $3 \times x + 5 = 3 \times 2 + 5 = 6 + 5 = 11$
\item $5 \times x – 9 = 5 \times 2 – 9 = 10 – 9 = 1$
\item $11 \neq 1$. Donc l'égalité $3 \times x + 5 = 5 \times x – 9$ est fausse pour $x = 2$
\end{enumerate}
\end{minipage}
\hspace{5mm}\vline\hspace{5mm}
\begin{minipage}{8cm}
Cette égalité est-elle vraie pour $x = 7$ ?
\begin{enumerate}
\item $3 \times x + 5 = 3 \times 7 + 5 = 21 + 5 = 26$
\item $5 \times x – 9 = 5 \times 7 – 9 = 35 – 9 = 26$
\item on trouve le même résultats donc l'égalité \\ $3 \times x + 5 = 5 \times x – 9$ est vraie pour $x = 7$
\end{enumerate}
\end{minipage}
\end{ex}

\section{La distributivité}

\subsection{Appliquer la distributivité}

\begin{meth}[Appliquer la distributivité au calcul mental]

\begin{center}
\og Calculer mentalement $24 \times 101$ ! On trouve $2424$ ! \\
Quelle méthode permet d'obtenir ce résultat rapidement ? \fg{}
\end{center}

\medskip
\noindent\grandscarreaux{\linewidth}{4.8cm}

\medskip
\noindent \textbf{Astuce :}

\medskip
\noindent \begin{minipage}{5cm}
$101 = 100 + 1$\\
$99 = 100 – 1$\\
$1010 = 1000 + 10$\\
$12 = 10 + 2$\\
$105 = 100 + 5$
\end{minipage}
\begin{minipage}{10cm}
On connaît des règles de calcul mental pour multiplier par $10$ par $100$, par $1000$, par $2$, par $5$, etc ...

\medskip
\noindent On décompose donc un des facteurs en somme ou différence formée de termes du type $10$, $100$, $1000$, $1$, $2$, $5$, ...
\end{minipage}
\end{meth}

\begin{meth}[Appliquer la distributivité au calcul mental]
Calculer astucieusement : $131 \times 13 + 131 \times 87$

\medskip
\noindent\grandscarreaux{\linewidth}{4cm}

\medskip
\noindent \textbf{Astuce : }On reconnaît un facteur commun pour appliquer la distributivité.
\end{meth}
