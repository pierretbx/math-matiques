\section{Rappel sur les volumes}

\vspace{-3mm}
\subsection{Unité de mesure}

\begin{defn}
L'unité principale d'un volume est le mètre cube, noté m$^3$, qui est le volume d'un cube de $1$m d'arête.

\medskip
\noindent On peut aussi exprimer un volume en litre. $1$L = $1$dm$^3$
\end{defn}

\medskip
On a les égalités :

\vspace{-2mm}
\begin{enumerate}
\begin{tabularx}{\linewidth}{p{4.5cm}*{3}{>{\arraybackslash}X}}
\item[•] $1$ m$^3$ = $1 000$ dm$^3$ = $1 000$ L&
\item[•] $1$ dm$^3$ = $1 000$ cm$^3$&
\item[•] $1$ L = $10$ dL&
\item[•] $1$ dL = $10$ cL\\
\end{tabularx}
\end{enumerate}

\vspace{-3mm}
\section{Prisme}

\vspace{-3mm}
\subsection{Définition et représentation}

\vspace{-2mm}
\begin{defn}
Un prisme droit est un solide dont :
\begin{enumerate}
\item[•] Les faces latérales sont des rectangles perpendiculaires aux bases.
\item[•] Les deux bases sont deux polygones parallèles et superposables ($2$ triangles, $2$ quadrilatères, etc..)
\end{enumerate}
\end{defn}

\begin{memento}[Représentation en perspective cavalière]
\begin{enumerate}
\item[•] Les droites parallèles restent parallèles.
\item[•] Les arêtes cachées sont représentées en pointillé.
\end{enumerate}
\end{memento}

\begin{rem}

\vspace{-4.5mm} \hspace{2.8cm} Le cube et le pavé droit sont des prismes droits particuliers.
\end{rem}

\vspace{-3mm}
\subsection{Patron}

\vspace{-2mm}
\begin{meth}
\begin{enumerate}
\item Compter le nombre total de faces (faces latérales et bases) du prisme.
\item Choisir une face latérale ou une base pour le premier tracé.
\item Faire à main levée le schéma d'un patron en repérant et en codant les longueurs égales.
\item Faire ce patron en construisant les faces en vraie forme et en vraie grandeur. Vérifier le patron.
\end{enumerate}
\end{meth}

\begin{ex}
Patron d'un prisme droit de $5$ cm de hauteur et dont les bases \\sont des triangles de dimensions $3$ cm, $3,5$ cm et $4$ cm.

\end{ex}

\section{Volume d'un prisme}

\vspace{-2mm}
\begin{prop}
Le volume d'un prisme droit s'obtient en multipliant l'aire de la base par la hauteur :

\medskip
\noindent \begin{minipage}{6cm}
$$V = A_B \times h_P$$
\end{minipage}
\begin{minipage}{10cm}
$V$ est le volume du solide, en unité de volume

$A_B$ est l'aire de la base, en unité d'aire

$h_P$ est la hauteur du prisme, en unité de longueur.
\end{minipage}
\end{prop}

\begin{rem}
Toutes les longueurs doivent être exprimées dans la même unité de mesure.
Dans la formule $V = AB \times h_P$ , $A_B$ est l'aire de la base du prisme (c'est un polygone).
(Aire : d'un triangle : $A = \dfrac{c \times h}{2}$, d'un rectangle : $A = L \times l$, d'un parallélogramme : $A = c \times h$ ...)
\end{rem}

\begin{ex}

\vspace{-5mm}
\noindent \begin{minipage}{5cm}
\begin{center}
\textbf{Exemple}
\end{center}

La base est un triangle
rectangle ;

\noindent Son aire est :
$A_{triangle} = \dfrac{c \times h_t}{2} $

soit $A_{triangle} = \dfrac{1,8 \times 3,6}{2}$

On a donc $A_B = 3,24$ cm$^2$.

\medskip
Le volume du prisme est :
$$V = A_B \times h_p$$

Soit $V = 3,24 \times 4 = 12,96$

\noindent 
Le volume du prisme droit est
$12,96$ cm$^3$.
\end{minipage}
\begin{minipage}{8cm}
\begin{center}
\textbf{Méthode}
\end{center}

\begin{enumerate}
\item On calcule l'aire de la base :
\begin{enumerate}
\item[•] repérer la forme de la base
\item[•] écrire la formule d'aire correspondante ;
\item[•] remplacer par les mesures connues ;
\item[•] effectuer le calcul.
\end{enumerate}
\item On calcule le volume du prisme droit :
\begin{enumerate}
\item[•] écrire la formule du volume d'un prisme
droit.
\item[•] remplacer par les nombres connus.
\end{enumerate}
\item On conclut avec les unités.
\end{enumerate}
\end{minipage}
