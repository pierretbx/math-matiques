\begin{ex}
Complète.

\begin{enumerate}[font=\color{exer1}]
\renewcommand{\arraystretch}{1.2}
\begin{tabularx}{\linewidth}{m{1.7cm}*{1}{>{\arraybackslash}X}m{0.5cm}m{2cm}*{1}{>{\arraybackslash}X}}
\item $3$m$^2$ $= $ & \dotfill cm$^2$&&
\item $700$cm$^2$ $= $ & \dotfill m$^2$\\
\item $5$m$^2$ $= $ & \dotfill dm$^2$&&
\item $6000$dm$^2$ $= $ & \dotfill m$^2$\\
\item $5$cm$^2$ $= $ & \dotfill mm$^2$&&
\item $400$mm$^2$ $= $ & \dotfill cm$^2$\\
\end{tabularx}
\end{enumerate}
\end{ex}

\begin{ex}
On donne les superficies suivantes :
\begin{enumerate}[font=\color{exer2}]
\renewcommand{\arraystretch}{0.7}
\begin{tabularx}{\linewidth}{*{2}{>{\arraybackslash}X}}

\item[•] Belle-île-en-mer : $\np{2300}$ ha \\
\item[•] Belle-île-en-mer : $\np{175000000}$m$^2$&
\item[•] Belle-île-en-mer : $\np{1160000}$dam$^2$\\
\end{tabularx}
\end{enumerate}
Range ces iles dans l'ordre décroissant de leur superficie.
\end{ex}
