\section*{Statistiques}

\begin{ex}
On a relevé les tailles, en cm, de $12$ élèves :
\[
145;\ 150;\ 148;\ 152;\ 150;\ 147;\ 155;\ 150;\ 148;\ 151;\ 149;\ 152.
\]
\begin{enumerate}
\item Donner l'effectif total.
\item Construire un tableau d'effectifs.
\item Calculer la fréquence de la valeur $150$.
\item Déterminer la médiane.
\end{enumerate}
\end{ex}

\begin{ex}
Les ventes d'une boulangerie sont données dans le tableau suivant.
\begin{center}
\begin{tabular}{|c|c|c|c|}
\hline
Produit & Baguettes & Croissants & Pains au chocolat\\
\hline
Effectif & $84$ & $36$ & $30$\\
\hline
\end{tabular}
\end{center}
\begin{enumerate}
\item Calculer l'effectif total.
\item Calculer les fréquences en pourcentage.
\item Représenter les résultats par un diagramme en bâtons.
\end{enumerate}
\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}

\begin{ex}
Calculer la moyenne des séries suivantes.
\begin{enumerate}
\item $8;\ 11;\ 13;\ 14;\ 14$
\item $2,5;\ 3;\ 4,5;\ 5$
\item $12;\ 7;\ 9;\ 16;\ 11;\ 17$
\end{enumerate}
\end{ex}
