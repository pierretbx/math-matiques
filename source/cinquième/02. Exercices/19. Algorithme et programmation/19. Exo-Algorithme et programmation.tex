\section*{Algorithme et programmation}

\begin{ex}
On considère le programme de calcul suivant.
\begin{center}
\begin{tabular}{|l|}
\hline
Choisir un nombre.\\
Ajouter $6$.\\
Multiplier par $3$.\\
Soustraire le nombre de départ.\\
\hline
\end{tabular}
\end{center}
\begin{enumerate}
\item Tester ce programme avec $2$, puis avec $10$.
\item Écrire l'expression obtenue avec un nombre de départ $x$.
\item Réduire cette expression.
\end{enumerate}
\end{ex}

\begin{ex}
Compléter l'algorithme permettant de calculer le périmètre d'un rectangle.
\begin{center}
\begin{tabular}{|l|}
\hline
Demander la longueur $L$.\\
Demander la largeur $\ell$.\\
Calculer $P=\ldots$\\
Afficher $P$.\\
\hline
\end{tabular}
\end{center}
\end{ex}
