\section*{Expressions littérales 2}

\begin{ex}
Réduire les expressions suivantes.
\begin{multicols}{2}
\begin{enumerate}
\item $3x+5x$
\item $7a-2a+4$
\item $5y+3-2y+8$
\item $4x+6+2x-1$
\end{enumerate}
\end{multicols}
\end{ex}

\begin{ex}
Développer les expressions suivantes.
\begin{multicols}{2}
\begin{enumerate}
\item $3(x+4)$
\item $5(2a-1)$
\item $7(3-y)$
\item $2(4x+6)$
\end{enumerate}
\end{multicols}
\end{ex}

\begin{ex}
Calculer la valeur de $A=4x+7$ et $B=3(2x-5)$ pour $x=0$, puis pour $x=3$.
\end{ex}

\begin{ex}
Un rectangle a pour longueur $x+5$ et pour largeur $x$.
\begin{enumerate}
\item Exprimer son périmètre en fonction de $x$.
\item Réduire l'expression obtenue.
\item Calculer ce périmètre pour $x=8$ cm.
\end{enumerate}
\end{ex}
