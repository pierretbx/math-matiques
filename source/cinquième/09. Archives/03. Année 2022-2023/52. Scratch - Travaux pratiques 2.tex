\blockvariable{mettre \selectmenu{résultat} à \ovaloperator{\ovalvariable{résultat}+\ovalnum{25}}}
\blockvariable{mettre \selectmenu{résultat} à \ovaloperator{\ovalvariable{résultat} * \ovalnum{-3}}}
\blocklook{dire \ovalvariable{résultat} pendant \ovalnum{6} secondes}
\end{scratch}
\end{minipage}
