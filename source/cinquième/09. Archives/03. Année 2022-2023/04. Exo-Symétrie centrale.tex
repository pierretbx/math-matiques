\item Les quadrilatères n°1 et n°3 sont symétriques par rapport au point \dots ~\dots ~
\item Les quadrilatère n°\dots ~\dots ~ est le symétrique du quadrilatère n°1 par la symétrie de centre $A$.
\item Le quadrilatère n°6 est l'image du quadrilatère n°1 par \dots ~\dots ~ \dots ~\dots ~ \dots ~\dots ~ \dots ~\dots ~

\end{enumerate}

\begin{exer3}

\begin{enumerate}[font=\color{exer3}]
\item Place un point $O$. Trace trois droites $(d_1)$,$(d_2)$ et $(d_3)$ passant par le point $O$.
\item Place un point $R$ sur $(d_1)$, un point $B$ sur $(d_2)$ et un point $E$ sur $(d_3)$.
\item En utilisant uniquement ton compas, place les points $m$, $U$ et $T$ pour que les triangles $MER$ et $BUT$ soient symétriques par rapport au point $O$.
\end{enumerate}
\end{exer3}
