\chapter{Cylindre et volume}

\section{Définition et représentation}

\noindent \begin{minipage}{11cm}
\begin{dft}
Un cylindre de révolution est un solide qui possède :
\begin{enumerate}
\item[•] Deux bases qui sont des disques superposables et
parallèles ;
\item[•] Une surface courbée appelée surface latérale.
\end{enumerate}
\end{dft}

\begin{rmq}
Sur la représentation en perspective cavalière :
\begin{enumerate}
\item[•] les disques sont \og écrasés \fg{}, ils se représentent par des \og ovales \fg{} .
\item[•] Les droites parallèles restent parallèles.
\end{enumerate}
\end{rmq}
\end{minipage}

\section{Patron du cylindre}

\noindent \begin{minipage}{8.5cm}
\begin{dft}
Un patron de cylindre est constitué :
\begin{enumerate}
\item[•] des deux disques
\item[•] d'un rectangle
\end{enumerate}
\end{dft}

\begin{rmq}
La largeur du rectangle est égale à la hauteur du cylindre ;
la longueur du rectangle est égale au périmètre de la base
\end{rmq}

\begin{meth}[Construction du patron d'un cylindre]
\begin{enumerate}
\item Calculer le périmètre du disque de base.
\item Tracer un rectangle de dimensions : 
\begin{enumerate}
\item[•] le périmètre du disque de base, 
\item[•] la hauteur du cylindre.
\end{enumerate}
\noindent\grandscarreaux{7.2cm}{1.6cm}
\end{enumerate}
\end{meth}
\end{minipage}
\begin{minipage}{7cm}

\end{center}

\begin{enumerate}
\setcounter{enumi}{2}

\item Pour tracer les deux disques, à l'extérieur du rectangle :
\begin{enumerate}
\item[•] Choisir les côtés ayant pour longueur le périmètre du disque de base
\item[•] Sur chacun de ces côtés, tracer un rayon du disque perpendiculaire à ce côté.
\item[•] Tracer, à partir de ces rayons, les deux cercles extérieurs au rectangle.
\end{enumerate}
\end{enumerate}

Réaliser un patron d'un cylindre de révolution de $7$cm de hauteur et de $2$cm de diamètre.

\end{ex}

\section{Volume du cylindre}

\begin{prop}
Le volume $\mathcal{V}$ d'un cylindre de rayon $r$ et de hauteur $h$ est obtenu grâce à la formule :
$$ \mathcal{V} = \pi \times r^2 \times h $$
\end{prop}

\begin{rmq}
On retrouve la formule du volume du prisme droit : $$\mathcal{V} = \mathcal{A}_{base} \times h$$
avec $\mathcal{A}_{base}$ l'aire du disque égale à $\pi \times r^2$
\end{rmq}

\begin{ex}
Calcule le volume du cylindre ci-contre, de $3$cm de rayon et de $7$cm de hauteur.

\end{ex}
