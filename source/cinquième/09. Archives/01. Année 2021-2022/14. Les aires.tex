\subsection{L'aire}

\begin{dft}
La surface d'une figure est la partie qui se trouve à l'intérieur de la figure.
\noindent L'aire est la mesure de la surface .
\end{dft}

\begin{meth}[Utilisation d'un tableau de conversion]

Convertir $28$m$^2$ en centimètre carré et $4,32$dm$^2$ en mètre carré. Attention, pour chaque unité, il y a deux colonnes.

\begin{rmq}
En agriculture, on utilise les unités agraires : l'hectare (ha), l'are (a) et le centiare (ca), pour calculer des superficies.
\end{rmq}

\begin{prop}[Aire d'un disque]
L'aire d'un disque de rayon $R$ est égal à :
$$\mathcal{A} = \pi \times R^2$$
\end{prop}

\begin{ex}
\begin{enumerate}
\item Calculer l'aire d'un disque de rayon $4$cm. Prendre $\pi \approx 3,14$.
