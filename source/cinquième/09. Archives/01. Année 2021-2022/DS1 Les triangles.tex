\paragraph{Exercice 3}: \hfill \textbf{6 points}

\noindent \begin{enumerate}
\item Construis un triangle isocèle $XYZ$ de sommet principal $Z$ tel que : $XZ=3,5$cm et $XY=6$cm.
\item Construis un triangle $TRS$ rectangle en $S$ tel que : $TS=7,2$cm et $SR=8,5$cm.
\item Construis un triangle $GLU$ rectangle en $L$ tel que : $LG=8$cm et $GU=10$cm.
\end{enumerate}

\paragraph{Exercice 4}: \hfill \textbf{5 points}

\noindent Un triangle a deux cotés dont les mesures sont $2$cm et $3$cm.
\begin{enumerate}
\item Donne une longueur possible du troisième coté
\item Il y a plusieurs possibilités pour la longueur de ce troisième coté mais Ninon affirme que toutes ces longueurs sont comprises entre deux nombres. Quels sont-ils ?
\end{enumerate}
