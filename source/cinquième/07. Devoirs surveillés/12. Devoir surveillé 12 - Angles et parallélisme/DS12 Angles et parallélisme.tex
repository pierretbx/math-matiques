\section*{Devoir surveillé 12 - Angles et parallélisme}
\addcontentsline{toc}{section}{Devoir surveillé 12 - Angles et parallélisme}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Angles]
Nommer deux angles alternes-internes et deux angles correspondants sur une figure donnée.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Parallélisme]
Deux angles alternes-internes mesurent chacun $64^\circ$. Que peut-on conclure sur les droites ? Justifier.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Calcul]
Deux droites parallèles sont coupées par une sécante. Un angle mesure $118^\circ$. Déterminer la mesure de l'angle adjacent et d'un angle correspondant.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
