\section*{Devoir surveillé 13 - Les triangles 2}
\addcontentsline{toc}{section}{Devoir surveillé 13 - Les triangles 2}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Construction]
Construire un triangle $ABC$ tel que $AB=6$ cm, $AC=4$ cm et $BC=5$ cm.
\medskip\noindent\grandscarreaux{\linewidth}{6cm}
\end{ex}
\begin{ex}[Angles]
Dans un triangle $RST$, $\widehat{R}=48^\circ$ et $\widehat{S}=73^\circ$. Calculer $\widehat{T}$.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Raisonner]
Un triangle isocèle a un angle au sommet de $42^\circ$. Calculer les deux autres angles.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
