\section*{Devoir surveillé 4 - Symétrie centrale}
\addcontentsline{toc}{section}{Devoir surveillé 4 - Symétrie centrale}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Construction]
Construire les symétriques des points $A$, $B$ et $C$ par rapport au point $O$.
\medskip\noindent\grandscarreaux{\linewidth}{6cm}
\end{ex}
\begin{ex}[Propriétés]
Dans une symétrie centrale de centre $O$, que peut-on dire de $O$ pour un point $M$ et son image $M'$ ?
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
\begin{ex}[Raisonner]
Expliquer pourquoi la symétrie centrale conserve les longueurs et les angles.
\medskip\noindent\grandscarreaux{\linewidth}{3cm}
\end{ex}
