\section*{Devoir surveillé 2 - Périmètre et aires}
\addcontentsline{toc}{section}{Devoir surveillé 2 - Périmètre et aires}

\noindent \textit{L'usage de la calculatrice est autorisé uniquement si le professeur l'indique.} \hfill \textit{Durée : 30 minutes}

\medskip
\noindent \textit{La rédaction, les justifications et la propreté seront prises en compte.}


\begin{ex}[Périmètre]
Calculer le périmètre d'un rectangle de longueur $8,4$ cm et de largeur $3,6$ cm, puis celui d'un carré de côté $5,5$ cm.
\medskip\noindent\grandscarreaux{\linewidth}{3.5cm}
\end{ex}
\begin{ex}[Aire]
Calculer l'aire d'un triangle de base $12$ cm et de hauteur $7$ cm, puis l'aire d'un disque de rayon $4$ cm.
\medskip\noindent\grandscarreaux{\linewidth}{4cm}
\end{ex}
\begin{ex}[Problème]
Un jardin rectangulaire mesure $18$ m sur $12$ m. Une allée de largeur $1$ m longe tout le bord intérieur. Calculer l'aire de l'allée.
\medskip\noindent\grandscarreaux{\linewidth}{5cm}
\end{ex}
