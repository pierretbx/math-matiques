\section*{Aire et périmètre \hspace{3cm} Sujet A}
\addcontentsline{toc}{section}{Aire et périmètre \hspace{3cm} Sujet A}

\noindent \textit{L'usage de la calculatrice est \textbf{autorisée}.} \hfill \textit{Durée : 30 minutes}

\paragraph{Exercice 1} : Donner votre définition du périmètre d'une figure géométrique. \hfill \textbf{2 points}

\paragraph{Exercice 2} : Complète.\hfill \textbf{6 points}

\begin{enumerate}
\renewcommand{\arraystretch}{1.5}
\begin{tabularx}{\linewidth}{m{1.7cm}*{1}{>{\arraybackslash}X}m{0.5cm}m{2cm}*{1}{>{\arraybackslash}X}}
\item $3,5$ hm $= $ & \dotfill m&&
\item $25$ cm$^2$ $= $ & \dotfill m$^2$\\
\item $22,3$ mm $= $ & \dotfill m&&
\item $3$ m$^2$ $= $ & \dotfill km$^2$\\
\item $7$ m $= $ & \dotfill cm&&
\item $5$ cm$^2$ $= $ & \dotfill dm$^2$\\
\end{tabularx}
\end{enumerate}

\paragraph{Exercice 3} : \hspace{6cm} \textbf{4 points}

\begin{enumerate}
\item Sur le parcours de santé ci-contre, repasse en vert les petits \\arc de cercle de même rayon et repasse en rouge les grands \\arcs de cercle de même rayon.

\item Calcule la longueur réelle du parcours de santé, au mètre près.
\end{enumerate}
