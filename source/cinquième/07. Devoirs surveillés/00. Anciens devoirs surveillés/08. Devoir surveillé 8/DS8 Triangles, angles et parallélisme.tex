\paragraph{Exercice 4} : \quad $ABC$ est un triangle tel que $\widehat{C} = 80$\degre et $\widehat{A} = 35$\degre. Calculer l'angle $\widehat{B}$. \hfill \textbf{5 points}

\medskip
\noindent \begin{minipage}{12cm}

\medskip
\noindent \textbf{On sait que :} \hspace{6.7mm}$\bullet$ \dotfill \medskip \medskip%$ABC$ est isocèle en $A$

\hspace{2.5cm} $\bullet$ \dotfill \medskip %$\widehat{C}= 80$\degre.

\medskip
\noindent \textbf{Propriété :} \dotfill \medskip \medskip

\dotfill \medskip%Dans un triangle isocèle, les deux angles à la base ont la même mesure. 
\end{minipage}
\hspace{3mm}
\begin{minipage}{4cm}
\begin{tikzpicture}[line cap=round,line join=round,>=triangle 45,x=1.0cm,y=1.0cm,scale= 0.7]
\clip(-1.8,-0.8) rectangle (4.8,4.);

\draw [line width=1.2pt] (4.,0.)-- (3.,3.);
\draw [line width=1.2pt] (3.,3.)-- (-1.,0.);
\begin{scriptsize}
\draw [color=blue] (-1.,0.)-- ++(-3.5pt,0 pt) -- ++(7.0pt,0 pt) ++(-3.5pt,-3.5pt) -- ++(0 pt,7.0pt);
\draw[color=blue] (-1.46,-0.27) node {\large{$A$}};
\draw [color=blue] (4.,0.)-- ++(-3.5pt,0 pt) -- ++(7.0pt,0 pt) ++(-3.5pt,-3.5pt) -- ++(0 pt,7.0pt);
\draw[color=blue] (4.32,-0.23) node {\large{$B$}};
\draw [color=blue] (3.,3.)-- ++(-3.5pt,0 pt) -- ++(7.0pt,0 pt) ++(-3.5pt,-3.5pt) -- ++(0 pt,7.0pt);
\draw[color=blue] (3.3,3.49) node {\large{$C$}};
\end{scriptsize}
\end{tikzpicture}
\end{minipage}

\medskip
\noindent \textbf{Conclusion :} 

\medskip
\noindent\grandscarreaux{\linewidth}{2.4cm}%$\widehat{B}$ et $\widehat{C}$ ont la même mesure. Donc : $$\widehat{B} = \widehat{C} = 80$$
%La mesure de l'angle $\widehat{B}$ est de $80$\degre .
