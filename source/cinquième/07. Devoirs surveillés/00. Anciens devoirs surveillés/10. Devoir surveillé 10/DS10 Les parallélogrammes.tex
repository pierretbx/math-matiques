\begin{enumerate}

\paragraph{Exercice 4} : \hfill \textbf{6 points}

\medskip
\noindent \textit{Pour chaque question}, construis d'abord une figure main levée, codée, sur ta feuille de copie puis \textbf{démontre} que c'est un parallélogramme.

\begin{enumerate}
\item VERT est un quadrilatère non croisé tel que $RT = VE$ et $VT = RE$.
\item BLEU est un quadrilatère non croisé tel que $\widehat{LBU}= \widehat{LEU}$ et $\widehat{BLE}= \widehat{BUE}$ 
\end{enumerate}
