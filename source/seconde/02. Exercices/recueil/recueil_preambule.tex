  \chapter*{Présentation}
  \addcontentsline{toc}{chapter}{Présentation}
  Ce recueil d'exercices est le résultat d'un travail partagé des enseignants de mathématiques du lycée. C'est projet commun, pensé pour accompagner nos élèves dans leur progression et dans la construction de bases solides.

  Au fil des pages, nous avons voulu proposer un parcours clair, cohérent et accessible, permettant à tous
  de développer des méthodes, de s'entraîner régulièrement et d'aborder plus sereinement la suite de
  leur scolarité.
  
   Ce recueil a été imaginé comme un tronc commun, garantissant une unité d'exigences et de préparation en vue de la première.

\newcommand{\ExtraireEltThierry}[3]{%
  \setcounter{ExtraireThierry}{1}%
  \@for\@termeE := #3 \do{%
    \ifthenelse{\theExtraireThierry=#2}{%
      \FPround{#1}{\@termeE{}}{1}%
      \stepcounter{ExtraireThierry}}%
    {\stepcounter{ExtraireThierry}}}}

\newcommand{\MyBezier}[3]{%
  \setcounter{NbreTripletsThierry}{0}%
  \listlenstore{NbreTripletsThierry}{#1}%
  \setcounter{NbreTripletsThierry}{(\theNbreTripletsThierry/3)-1}%
  \ExtraireEltThierry{\Xa}{1}{#1}%
  \ExtraireEltThierry{\Ya}{2}{#1}%
  \ExtraireEltThierry{\Yprimea}{3}{#1}%
  \multido{\n=4+3}{\theNbreTripletsThierry}{%
    \setcounter{TempoThierry}{\n}%
    \ExtraireEltThierry{\Xb}{\theTempoThierry}{#1}%
    \stepcounter{TempoThierry}%
    \ExtraireEltThierry{\Yb}{\theTempoThierry}{#1}%
    \stepcounter{TempoThierry}%
    \ExtraireEltThierry{\Yprimeb}{\theTempoThierry}{#1}%
    \FPeval\LgTangentea{(1+(Yprimea)*(Yprimea))^(0.5)}%
    \FPeval\LgTangenteb{(1+(Yprimeb)*(Yprimeb))^(0.5)}%
    \FPeval\Xc{Xa+#2*0.5*(Xb-Xa)/(LgTangentea)}%
    \FPeval\Yc{Ya+#2*0.5*(Xb-Xa)*(Yprimea)/(LgTangentea)}%
    \FPeval\Xd{Xb-#2*0.5*(Xb-Xa)/(LgTangenteb)}%
    \FPeval\Yd{Yb-#2*0.5*(Xb-Xa)*(Yprimeb)/(LgTangenteb)}%
    \psbezier[showpoints=#3](\Xa,\Ya)(\Xc,\Yc)(\Xd,\Yd)(\Xb,\Yb)%
    \FPset\Xa\Xb
    \FPset\Ya\Yb
    \FPset\Yprimea\Yprimeb}}
