\exo
Comparer les nombres suivants sans utiliser la calculatrice.
\setlength{\columnseprule}{0pt}
\begin{multicols}{2}
\begin{enumerate}
\item $\sqrt{\dfrac{1}{2}}$ et $\sqrt{\dfrac{1}{3}}$
\item $\sqrt{2,001}$ et $\sqrt{2,01}$
\item $\sqrt{\pi}$ et $\sqrt{3}$
\end{enumerate}
\end{multicols}

\exo

\begin{multicols}{3}
\begin{enumerate}
\item $\sqrt{x}=2$
\item $\sqrt{x}=3$
\item $-\sqrt{x}=-\dfrac{3}{4}$
\end{enumerate}
\end{multicols}
